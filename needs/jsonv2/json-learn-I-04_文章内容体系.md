## 4. Go 数据加载实现框架

```go
// data/articles.go
package data

import (
    "encoding/json"
    "os"
    "sort"
    "sync"
)

var (
    articles     []Article
    articleMap   map[string]*Article
    articlesOnce sync.Once
)

// LoadArticles 从 JSON 文件加载文章元数据（应用启动时调用一次）
func LoadArticles() {
    articlesOnce.Do(func() {
        file, _ := os.ReadFile("data/articles.json")
        json.Unmarshal(file, &articles)

        // 按 sort_order 排序
        sort.Slice(articles, func(i, j int) bool {
            return articles[i].SortOrder < articles[j].SortOrder
        })

        // 构建 slug -> Article 映射
        articleMap = make(map[string]*Article, len(articles))
        for i := range articles {
            articleMap[articles[i].Slug] = &articles[i]
        }
    })
}

// GetAllArticles 返回全部文章列表
func GetAllArticles() []Article { return articles }

// FindBySlug 按 slug 查找单篇文章
func FindBySlug(slug string) *Article { return articleMap[slug] }

// GetAdjacent 获取上下篇
func GetAdjacent(slug string) (*Article, *Article) {
    for i, a := range articles {
        if a.Slug == slug {
            var prev, next *Article
            if i > 0 { prev = &articles[i-1] }
            if i < len(articles)-1 { next = &articles[i+1] }
            return prev, next
        }
    }
    return nil, nil
}

// GetRelated 获取相关文章（同分类或同级别，排除自身）
func GetRelated(category, level, excludeSlug string, limit int) []Article {
    var results []Article
    for _, a := range articles {
        if a.Slug == excludeSlug { continue }
        if a.Category == category || a.Level == level {
            results = append(results, a)
            if len(results) >= limit { break }
        }
    }
    return results
}

// LoadArticleContent 加载 Markdown 文件并渲染为 HTML
func LoadArticleContent(slug, lang string) string {
    // 尝试加载本地化版本
    path := "content/" + lang + "/" + slug + ".md"
    if _, err := os.Stat(path); os.IsNotExist(err) {
        path = "content/zh/" + slug + ".md" // fallback 到中文
    }
    content, _ := os.ReadFile(path)
    // 使用 goldmark 或 blackfriday 渲染 Markdown → HTML
    return renderMarkdown(string(content))
}
```
