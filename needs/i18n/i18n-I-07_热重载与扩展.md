# 热重载与扩展方案

## internal/i18n/watcher.go（开发环境热重载）

```go
package i18n

import (
	"log"
	"path/filepath"
	"time"

	"github.com/fsnotify/fsnotify"
)

// Watcher 监听 locales 目录变化，自动触发热重载
type Watcher struct {
	manager    *Manager
	dir        string
	fsWatcher  *fsnotify.Watcher
	debounceMs int
}

func NewWatcher(manager *Manager, localesDir string) (*Watcher, error) {
	fw, err := fsnotify.NewWatcher()
	if err != nil {
		return nil, err
	}
	return &Watcher{
		manager:    manager,
		dir:        localesDir,
		fsWatcher:  fw,
		debounceMs: 300,
	}, nil
}

// Start 启动文件监听（在 goroutine 中运行）
func (w *Watcher) Start() error {
	// 监听所有子目录
	if err := filepath.WalkDir(w.dir, func(path string, d interface{ IsDir() bool }, err error) error {
		if err != nil || !d.IsDir() {
			return err
		}
		return w.fsWatcher.Add(path)
	}); err != nil {
		return err
	}

	go w.watch()
	log.Printf("[i18n] Watching %s for changes...", w.dir)
	return nil
}

func (w *Watcher) watch() {
	timer := time.NewTimer(0)
	timer.Stop()

	for {
		select {
		case event, ok := <-w.fsWatcher.Events:
			if !ok {
				return
			}
			if filepath.Ext(event.Name) == ".json" {
				// 防抖：300ms 内多次变更只触发一次重载
				timer.Reset(time.Duration(w.debounceMs) * time.Millisecond)
			}
		case <-timer.C:
			log.Println("[i18n] Detected changes, reloading...")
			if err := w.manager.Reload(); err != nil {
				log.Printf("[i18n] Reload error: %v", err)
			} else {
				log.Println("[i18n] Reloaded successfully")
			}
		case err, ok := <-w.fsWatcher.Errors:
			if !ok {
				return
			}
			log.Printf("[i18n] Watcher error: %v", err)
		}
	}
}

func (w *Watcher) Close() error {
	return w.fsWatcher.Close()
}
```

## cmd/server/main.go（完整启动入口）

```go
package main

import (
	"log"
	"os"

	"yourproject/internal/i18n"
	"yourproject/router"
)

func main() {
	// i18n 配置
	cfg := i18n.Config{
		LocalesDir:  "./locales",
		DefaultLang: "en-US",
		Fallbacks: map[string][]string{
			"zh-TW": {"zh-CN", "en-US"},
			"zh-HK": {"zh-CN", "en-US"},
		},
	}

	// 初始化 i18n Manager
	manager, err := i18n.NewManager(cfg)
	if err != nil {
		log.Fatalf("Failed to initialize i18n: %v", err)
	}

	// 开发模式启动热重载
	if os.Getenv("GIN_MODE") != "release" {
		watcher, err := i18n.NewWatcher(manager, cfg.LocalesDir)
		if err != nil {
			log.Printf("Warning: could not start i18n watcher: %v", err)
		} else {
			if err := watcher.Start(); err != nil {
				log.Printf("Warning: could not start watching: %v", err)
			}
			defer watcher.Close()
		}
	}

	// 设置路由
	r := router.Setup(manager)

	log.Println("Server starting on :8080")
	log.Printf("Supported languages: %v", manager.SupportedLangs())
	if err := r.Run(":8080"); err != nil {
		log.Fatalf("Server error: %v", err)
	}
}
```

## 扩展：新增一门语言的流程

```bash
# 1. 创建目录
mkdir locales/fr-FR

# 2. 复制英文文件作为模板
cp locales/en-US/*.json locales/fr-FR/

# 3. 翻译 JSON 内容（可对接翻译 API）
# vim locales/fr-FR/common.json ...

# 4. 重启服务（或等待热重载）
# GIN_MODE=debug go run ./cmd/server

# 5. 测试
curl http://localhost:8080/ -H "Accept-Language: fr-FR"
# 或
curl "http://localhost:8080/?lang=fr-FR"
```

## 扩展：新增一个命名空间的流程

```bash
# 1. 在所有语言目录创建对应文件
touch locales/en-US/settings.json
touch locales/zh-CN/settings.json

# 2. 填写翻译内容
# locales/en-US/settings.json
{
  "page": { "title": "Settings" },
  "profile": { "save": "Save Profile" }
}

# 3. 在模板中直接使用（热重载自动生效）
# {{ .T.T "settings.page.title" }}
# {{ .T.T "settings.profile.save" }}
```

## 生产环境建议

| 场景 | 建议 |
|------|------|
| 翻译文件变更 | 通过 CI/CD 重新部署，不使用热重载 |
| 开发环境 | 启用 fsnotify 热重载，提升效率 |
| 多实例部署 | 翻译文件打包进镜像，无状态 |
| key 审计 | 定期运行脚本对比各语言 key 完整性 |

## key 完整性检查脚本（可选）

```bash
#!/bin/bash
# check-i18n.sh: 检查所有语言的 key 是否与 en-US 一致

BASE_LANG="en-US"
BASE_KEYS=$(find locales/$BASE_LANG -name "*.json" -exec cat {} \; | jq -r 'paths(scalars) | join(".")' | sort)

for lang_dir in locales/*/; do
    lang=$(basename "$lang_dir")
    [ "$lang" = "$BASE_LANG" ] && continue

    LANG_KEYS=$(find locales/$lang -name "*.json" -exec cat {} \; | jq -r 'paths(scalars) | join(".")' | sort)
    MISSING=$(comm -23 <(echo "$BASE_KEYS") <(echo "$LANG_KEYS"))

    if [ -n "$MISSING" ]; then
        echo "❌ $lang missing keys:"
        echo "$MISSING" | head -20
    else
        echo "✅ $lang: all keys present"
    fi
done
```
