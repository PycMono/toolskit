# i18n 核心引擎代码

## internal/i18n/manager.go

```go
package i18n

import (
	"fmt"
	"sync"
)

// Manager 是 i18n 的核心管理器，负责管理所有语言和命名空间
type Manager struct {
	mu          sync.RWMutex
	defaultLang string
	fallbacks   map[string][]string            // lang -> fallback chain
	messages    map[string]map[string]Messages // lang -> namespace -> messages
	loader      *Loader
}

// Messages 是命名空间内的扁平化翻译 map
// key 支持点分路径，如 "button.submit"
type Messages map[string]string

// Config 是管理器配置
type Config struct {
	LocalesDir  string
	DefaultLang string
	Fallbacks   map[string][]string // e.g. {"zh-TW": ["zh-CN", "en-US"]}
}

// NewManager 创建并初始化管理器
func NewManager(cfg Config) (*Manager, error) {
	m := &Manager{
		defaultLang: cfg.DefaultLang,
		fallbacks:   cfg.Fallbacks,
		messages:    make(map[string]map[string]Messages),
		loader:      NewLoader(cfg.LocalesDir),
	}
	if m.fallbacks == nil {
		m.fallbacks = make(map[string][]string)
	}
	if err := m.load(); err != nil {
		return nil, err
	}
	return m, nil
}

// load 加载所有语言的所有命名空间
func (m *Manager) load() error {
	data, err := m.loader.LoadAll()
	if err != nil {
		return fmt.Errorf("i18n: failed to load locales: %w", err)
	}
	m.mu.Lock()
	defer m.mu.Unlock()
	m.messages = data
	return nil
}

// Reload 热重载（供 watcher 调用）
func (m *Manager) Reload() error {
	return m.load()
}

// Translator 返回指定语言的翻译器
func (m *Manager) Translator(lang string) *Translator {
	chain := m.buildFallbackChain(lang)
	return &Translator{
		manager: m,
		lang:    lang,
		chain:   chain,
	}
}

// buildFallbackChain 构建 fallback 语言链
func (m *Manager) buildFallbackChain(lang string) []string {
	chain := []string{lang}
	if fallbacks, ok := m.fallbacks[lang]; ok {
		chain = append(chain, fallbacks...)
	}
	if lang != m.defaultLang {
		chain = append(chain, m.defaultLang)
	}
	// 去重
	seen := make(map[string]bool)
	result := make([]string, 0, len(chain))
	for _, l := range chain {
		if !seen[l] {
			seen[l] = true
			result = append(result, l)
		}
	}
	return result
}

// lookup 在 fallback 链中查找 key
func (m *Manager) lookup(chain []string, namespace, key string) (string, bool) {
	m.mu.RLock()
	defer m.mu.RUnlock()
	for _, lang := range chain {
		if nsMap, ok := m.messages[lang]; ok {
			if ns, ok := nsMap[namespace]; ok {
				if val, ok := ns[key]; ok {
					return val, true
				}
			}
		}
	}
	return "", false
}

// SupportedLangs 返回已加载的语言列表
func (m *Manager) SupportedLangs() []string {
	m.mu.RLock()
	defer m.mu.RUnlock()
	langs := make([]string, 0, len(m.messages))
	for lang := range m.messages {
		langs = append(langs, lang)
	}
	return langs
}
```

## internal/i18n/loader.go

```go
package i18n

import (
	"encoding/json"
	"fmt"
	"os"
	"path/filepath"
	"strings"
)

// Loader 负责从磁盘加载语言文件
type Loader struct {
	dir string
}

func NewLoader(dir string) *Loader {
	return &Loader{dir: dir}
}

// LoadAll 遍历 locales 目录，加载所有语言和命名空间
// 返回结构: lang -> namespace -> Messages
func (l *Loader) LoadAll() (map[string]map[string]Messages, error) {
	result := make(map[string]map[string]Messages)

	// 遍历语言目录（第一层）
	langEntries, err := os.ReadDir(l.dir)
	if err != nil {
		return nil, fmt.Errorf("cannot read locales dir %q: %w", l.dir, err)
	}

	for _, langEntry := range langEntries {
		if !langEntry.IsDir() {
			continue
		}
		lang := langEntry.Name()
		langDir := filepath.Join(l.dir, lang)

		nsMap, err := l.loadLang(langDir)
		if err != nil {
			return nil, fmt.Errorf("failed to load lang %q: %w", lang, err)
		}
		result[lang] = nsMap
	}
	return result, nil
}

// loadLang 加载单个语言目录中的所有命名空间
func (l *Loader) loadLang(langDir string) (map[string]Messages, error) {
	nsMap := make(map[string]Messages)

	files, err := os.ReadDir(langDir)
	if err != nil {
		return nil, err
	}

	for _, f := range files {
		if f.IsDir() || filepath.Ext(f.Name()) != ".json" {
			continue
		}
		ns := strings.TrimSuffix(f.Name(), ".json")
		filePath := filepath.Join(langDir, f.Name())

		msgs, err := l.loadFile(filePath)
		if err != nil {
			return nil, fmt.Errorf("failed to load %q: %w", filePath, err)
		}
		nsMap[ns] = msgs
	}
	return nsMap, nil
}

// loadFile 加载单个 JSON 文件，支持嵌套结构，展开为点分 key
func (l *Loader) loadFile(path string) (Messages, error) {
	data, err := os.ReadFile(path)
	if err != nil {
		return nil, err
	}

	var raw map[string]interface{}
	if err := json.Unmarshal(data, &raw); err != nil {
		return nil, fmt.Errorf("invalid JSON in %q: %w", path, err)
	}

	msgs := make(Messages)
	flatten("", raw, msgs)
	return msgs, nil
}

// flatten 将嵌套 JSON 展开为点分 key
// {"button": {"submit": "Submit"}} → {"button.submit": "Submit"}
func flatten(prefix string, obj map[string]interface{}, out Messages) {
	for k, v := range obj {
		fullKey := k
		if prefix != "" {
			fullKey = prefix + "." + k
		}
		switch val := v.(type) {
		case string:
			out[fullKey] = val
		case map[string]interface{}:
			flatten(fullKey, val, out)
		default:
			// 非字符串值转为字符串
			out[fullKey] = fmt.Sprintf("%v", val)
		}
	}
}
```

## internal/i18n/translator.go

```go
package i18n

import (
	"fmt"
	"strings"
)

// Translator 是请求级别的翻译器（轻量，不持有锁）
type Translator struct {
	manager *Manager
	lang    string
	chain   []string // fallback 链
}

// T 翻译 key（格式: "namespace.key" 或 "namespace.section.key"）
// 未找到时返回 key 本身
func (t *Translator) T(key string) string {
	ns, subKey := splitKey(key)
	if val, ok := t.manager.lookup(t.chain, ns, subKey); ok {
		return val
	}
	return key
}

// TF 带 fmt.Sprintf 插值的翻译
// 用法: TF("common.hello", "World") → "Hello, World!"
// JSON: "hello": "Hello, %s!"
func (t *Translator) TF(key string, args ...interface{}) string {
	val := t.T(key)
	if len(args) == 0 {
		return val
	}
	return fmt.Sprintf(val, args...)
}

// TN 复数翻译
// JSON: "items": {"one": "%d item", "other": "%d items"}
// 展开后: "items.one", "items.other"
func (t *Translator) TN(key string, n int) string {
	pluralKey := key + ".other"
	if n == 1 {
		pluralKey = key + ".one"
	}
	val := t.T(pluralKey)
	return fmt.Sprintf(val, n)
}

// TWith 指定命名空间翻译（不使用自动拆分）
func (t *Translator) TWith(namespace, key string) string {
	if val, ok := t.manager.lookup(t.chain, namespace, key); ok {
		return val
	}
	return namespace + "." + key
}

// Lang 返回当前语言
func (t *Translator) Lang() string {
	return t.lang
}

// IsRTL 判断是否是从右到左语言
func (t *Translator) IsRTL() bool {
	rtlLangs := map[string]bool{
		"ar": true, "ar-SA": true,
		"he": true, "he-IL": true,
		"fa": true, "fa-IR": true,
		"ur": true, "ur-PK": true,
	}
	return rtlLangs[t.lang]
}

// splitKey 将 "namespace.key.subkey" 拆分为 namespace 和 rest
// "common.button.submit" → ("common", "button.submit")
// "common.title"         → ("common", "title")
func splitKey(key string) (namespace, subKey string) {
	idx := strings.Index(key, ".")
	if idx == -1 {
		return key, ""
	}
	return key[:idx], key[idx+1:]
}
```

## internal/i18n/detector.go

```go
package i18n

import (
	"net/http"
	"strings"
)

// Detector 检测请求中的语言偏好
type Detector struct {
	supported   map[string]bool
	defaultLang string
}

func NewDetector(supported []string, defaultLang string) *Detector {
	m := make(map[string]bool, len(supported))
	for _, l := range supported {
		m[l] = true
	}
	return &Detector{supported: m, defaultLang: defaultLang}
}

// Detect 按优先级检测语言: URL param > Cookie > Accept-Language > default
func (d *Detector) Detect(r *http.Request) string {
	// 1. URL query param: ?lang=zh-CN
	if lang := r.URL.Query().Get("lang"); lang != "" {
		if l := d.normalize(lang); l != "" {
			return l
		}
	}

	// 2. Cookie: lang=zh-CN
	if cookie, err := r.Cookie("lang"); err == nil {
		if l := d.normalize(cookie.Value); l != "" {
			return l
		}
	}

	// 3. Accept-Language header
	if lang := d.parseAcceptLanguage(r.Header.Get("Accept-Language")); lang != "" {
		return lang
	}

	return d.defaultLang
}

// normalize 检查语言是否受支持，并返回规范化的 code
func (d *Detector) normalize(lang string) string {
	// 精确匹配
	if d.supported[lang] {
		return lang
	}
	// 尝试主语言匹配: "zh-TW" → "zh-CN"
	base := strings.SplitN(lang, "-", 2)[0]
	for l := range d.supported {
		if strings.HasPrefix(l, base+"-") || l == base {
			return l
		}
	}
	return ""
}

// parseAcceptLanguage 解析 Accept-Language header，按 q 值排序
func (d *Detector) parseAcceptLanguage(header string) string {
	if header == "" {
		return ""
	}
	// 简化解析：取第一个受支持的语言
	parts := strings.Split(header, ",")
	for _, part := range parts {
		lang := strings.TrimSpace(strings.SplitN(part, ";", 2)[0])
		if l := d.normalize(lang); l != "" {
			return l
		}
	}
	return ""
}
```
