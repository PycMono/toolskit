# 验收标准与测试用例

## internal/i18n/manager_test.go

```go
package i18n_test

import (
	"os"
	"path/filepath"
	"testing"

	"yourproject/internal/i18n"
)

// setupTestLocales 创建测试用的临时 locales 目录
func setupTestLocales(t *testing.T) string {
	t.Helper()
	dir := t.TempDir()

	// en-US/common.json
	enDir := filepath.Join(dir, "en-US")
	os.MkdirAll(enDir, 0755)
	os.WriteFile(filepath.Join(enDir, "common.json"), []byte(`{
		"button": {"submit": "Submit", "cancel": "Cancel"},
		"notification": {
			"count": {"one": "%d notification", "other": "%d notifications"}
		}
	}`), 0644)

	// en-US/auth.json
	os.WriteFile(filepath.Join(enDir, "auth.json"), []byte(`{
		"login": {"title": "Sign In", "welcome": "Welcome, %s!"}
	}`), 0644)

	// zh-CN/common.json
	zhDir := filepath.Join(dir, "zh-CN")
	os.MkdirAll(zhDir, 0755)
	os.WriteFile(filepath.Join(zhDir, "common.json"), []byte(`{
		"button": {"submit": "提交", "cancel": "取消"},
		"notification": {
			"count": {"one": "%d 条通知", "other": "%d 条通知"}
		}
	}`), 0644)

	return dir
}

func TestBasicTranslation(t *testing.T) {
	dir := setupTestLocales(t)
	m, err := i18n.NewManager(i18n.Config{
		LocalesDir:  dir,
		DefaultLang: "en-US",
	})
	if err != nil {
		t.Fatalf("NewManager: %v", err)
	}

	tr := m.Translator("en-US")
	got := tr.T("common.button.submit")
	if got != "Submit" {
		t.Errorf("T(%q) = %q, want %q", "common.button.submit", got, "Submit")
	}
}

func TestChineseTranslation(t *testing.T) {
	dir := setupTestLocales(t)
	m, _ := i18n.NewManager(i18n.Config{LocalesDir: dir, DefaultLang: "en-US"})

	tr := m.Translator("zh-CN")
	got := tr.T("common.button.submit")
	if got != "提交" {
		t.Errorf("got %q, want %q", got, "提交")
	}
}

func TestFallback(t *testing.T) {
	dir := setupTestLocales(t)
	m, _ := i18n.NewManager(i18n.Config{
		LocalesDir:  dir,
		DefaultLang: "en-US",
		Fallbacks:   map[string][]string{"zh-TW": {"zh-CN", "en-US"}},
	})

	// zh-TW 不存在 auth 翻译，应 fallback 到 en-US
	tr := m.Translator("zh-TW")
	got := tr.T("auth.login.title")
	if got != "Sign In" {
		t.Errorf("fallback got %q, want %q", got, "Sign In")
	}
}

func TestMissingKey(t *testing.T) {
	dir := setupTestLocales(t)
	m, _ := i18n.NewManager(i18n.Config{LocalesDir: dir, DefaultLang: "en-US"})

	tr := m.Translator("en-US")
	key := "common.nonexistent.key"
	got := tr.T(key)
	// 未找到时返回 key 本身
	if got != key {
		t.Errorf("missing key got %q, want key itself %q", got, key)
	}
}

func TestInterpolation(t *testing.T) {
	dir := setupTestLocales(t)
	m, _ := i18n.NewManager(i18n.Config{LocalesDir: dir, DefaultLang: "en-US"})

	tr := m.Translator("en-US")
	got := tr.TF("auth.login.welcome", "Alice")
	if got != "Welcome, Alice!" {
		t.Errorf("TF got %q, want %q", got, "Welcome, Alice!")
	}
}

func TestPlural(t *testing.T) {
	dir := setupTestLocales(t)
	m, _ := i18n.NewManager(i18n.Config{LocalesDir: dir, DefaultLang: "en-US"})
	tr := m.Translator("en-US")

	tests := []struct {
		n    int
		want string
	}{
		{1, "1 notification"},
		{5, "5 notifications"},
	}
	for _, tt := range tests {
		got := tr.TN("common.notification.count", tt.n)
		if got != tt.want {
			t.Errorf("TN(n=%d) got %q, want %q", tt.n, got, tt.want)
		}
	}
}

func TestConcurrentAccess(t *testing.T) {
	dir := setupTestLocales(t)
	m, _ := i18n.NewManager(i18n.Config{LocalesDir: dir, DefaultLang: "en-US"})

	// 并发读取不应 panic 或 data race
	done := make(chan struct{})
	for i := 0; i < 100; i++ {
		go func() {
			tr := m.Translator("zh-CN")
			_ = tr.T("common.button.submit")
			done <- struct{}{}
		}()
	}
	for i := 0; i < 100; i++ {
		<-done
	}
}
```

## internal/i18n/detector_test.go

```go
package i18n_test

import (
	"net/http"
	"testing"

	"yourproject/internal/i18n"
)

func TestDetect_URLParam(t *testing.T) {
	d := i18n.NewDetector([]string{"en-US", "zh-CN"}, "en-US")
	r, _ := http.NewRequest("GET", "/?lang=zh-CN", nil)
	got := d.Detect(r)
	if got != "zh-CN" {
		t.Errorf("URL param: got %q, want zh-CN", got)
	}
}

func TestDetect_Cookie(t *testing.T) {
	d := i18n.NewDetector([]string{"en-US", "zh-CN", "ja-JP"}, "en-US")
	r, _ := http.NewRequest("GET", "/", nil)
	r.AddCookie(&http.Cookie{Name: "lang", Value: "ja-JP"})
	got := d.Detect(r)
	if got != "ja-JP" {
		t.Errorf("Cookie: got %q, want ja-JP", got)
	}
}

func TestDetect_AcceptLanguage(t *testing.T) {
	d := i18n.NewDetector([]string{"en-US", "zh-CN", "ja-JP"}, "en-US")
	r, _ := http.NewRequest("GET", "/", nil)
	r.Header.Set("Accept-Language", "ja-JP,en-US;q=0.9")
	got := d.Detect(r)
	if got != "ja-JP" {
		t.Errorf("Accept-Language: got %q, want ja-JP", got)
	}
}

func TestDetect_Default(t *testing.T) {
	d := i18n.NewDetector([]string{"en-US", "zh-CN"}, "en-US")
	r, _ := http.NewRequest("GET", "/", nil)
	got := d.Detect(r)
	if got != "en-US" {
		t.Errorf("default: got %q, want en-US", got)
	}
}

func TestDetect_Priority(t *testing.T) {
	// URL param 优先级高于 Cookie
	d := i18n.NewDetector([]string{"en-US", "zh-CN", "ja-JP"}, "en-US")
	r, _ := http.NewRequest("GET", "/?lang=zh-CN", nil)
	r.AddCookie(&http.Cookie{Name: "lang", Value: "ja-JP"})
	got := d.Detect(r)
	if got != "zh-CN" {
		t.Errorf("priority: got %q, want zh-CN (URL param wins)", got)
	}
}
```

## 验收清单

### 功能验收

- [ ] 所有命名空间 JSON 文件启动时正确加载
- [ ] 嵌套 JSON 正确展平为点分 key
- [ ] `T("ns.key")` 返回正确翻译
- [ ] `TF("ns.key", args...)` 正确插值（%s / %d / %f）
- [ ] `TN("ns.key", n)` 正确处理单数/复数
- [ ] Fallback 链按配置顺序生效
- [ ] 未找到 key 时返回 key 本身（不 panic）
- [ ] URL `?lang=` 参数优先级最高
- [ ] Cookie `lang` 优先级次之
- [ ] `Accept-Language` header 兜底
- [ ] 语言切换后 Cookie 正确持久化（30天）
- [ ] 模板中 `{{ .T.T "key" }}` 正确渲染
- [ ] 模板中 `{{ .T.TF "key" .Var }}` 正确插值
- [ ] 模板中 `{{ .T.TN "key" .Count }}` 正确复数
- [ ] `Content-Language` 响应头正确设置
- [ ] RTL 语言正确输出 `dir="rtl"` 属性

### 性能验收

- [ ] 启动加载 < 100ms（1000 个 key 以内）
- [ ] 翻译查找 < 1μs（map 查找，无锁读路径）
- [ ] 热重载不影响正在处理的请求（RWMutex 保护）
- [ ] 100 并发读取无 data race（`go test -race`）

### 扩展性验收

- [ ] 新增语言：只需新建目录和文件，无需改代码
- [ ] 新增命名空间：只需新建 JSON 文件，无需改代码
- [ ] 热重载：修改 JSON 后 < 500ms 自动生效（开发环境）
- [ ] key 完整性脚本可检测缺失翻译

### 运行测试命令

```bash
# 运行所有测试
go test ./internal/i18n/... -v

# 竞争检测
go test ./internal/i18n/... -race

# 基准测试
go test ./internal/i18n/... -bench=. -benchmem

# 覆盖率报告
go test ./internal/i18n/... -coverprofile=coverage.out
go tool cover -html=coverage.out
```
