```javascript
// static/js/sms-active.js

let activeOrderWSMap = {};   // { orderId: WebSocket }

async function loadActiveOrders() {
  const resp = await fetch('/api/sms/orders/active');
  const data = await resp.json();

  const orders = data.orders || [];
  const list   = document.getElementById('active-list');
  const empty  = document.getElementById('active-empty');

  if (orders.length === 0) {
    empty.style.display = 'flex';
    return;
  }

  empty.style.display = 'none';

  orders.forEach(order => {
    const card = buildActiveOrderCard(order);
    list.appendChild(card);
    connectActiveOrderWS(order.id);
  });
}

function buildActiveOrderCard(order) {
  const card = document.createElement('div');
  card.className = 'active-order-card';
  card.id = `active-card-${order.id}`;

  const expiry   = new Date(order.expires_at);
  const secLeft  = Math.floor((expiry - Date.now()) / 1000);

  card.innerHTML = `
    <div class="active-card__header">
      <img src="/static/icons/sms/${order.product.toLowerCase()}.png" alt="${order.product}"
           class="active-service-icon" onerror="this.src='/static/icons/sms/default.png'">
      <div class="active-card__info">
        <span class="active-card__service">${order.product}</span>
        <span class="active-card__country">${order.country}</span>
      </div>
      <div class="active-card__timer" id="timer-${order.id}">
        ${formatTime(secLeft)}
      </div>
    </div>

    <div class="active-card__number">
      <span class="active-number-text">${formatPhoneNumber(order.phone)}</span>
      <button onclick="copyActiveNumber('${order.phone}', this)" class="copy-btn-sm">复制</button>
    </div>

    <!-- OTP 区域（初始隐藏）-->
    <div class="active-otp" id="active-otp-${order.id}" style="display:none">
      <span class="active-otp__code" id="active-otp-code-${order.id}">—</span>
      <button onclick="copyActiveOTP('${order.id}')" class="copy-btn-sm copy-btn-sm--green">
        复制验证码
      </button>
    </div>

    <!-- 等待指示器 -->
    <div class="active-waiting" id="active-waiting-${order.id}">
      <div class="waiting-dots-sm">
        <span></span><span></span><span></span>
      </div>
      <span>等待验证码...</span>
    </div>

    <div class="active-card__actions">
      <button onclick="cancelActiveOrder('${order.id}')" class="active-btn active-btn--cancel">取消</button>
      <button onclick="finishActiveOrder('${order.id}')" class="active-btn active-btn--finish"
              id="finish-btn-${order.id}" style="display:none">完成</button>
    </div>
  `;

  // 启动倒计时
  startActiveTimer(order.id, secLeft);
  return card;
}

function connectActiveOrderWS(orderId) {
  if (activeOrderWSMap[orderId]) return;

  const proto = location.protocol === 'https:' ? 'wss:' : 'ws:';
  const ws    = new WebSocket(`${proto}//${location.host}/ws/sms/${orderId}`);

  ws.onmessage = (e) => {
    const msg = JSON.parse(e.data);
    if (msg.type === 'sms' && msg.sms) {
      const otp = msg.sms.code || msg.sms.text?.match(/\b(\d{4,8})\b/)?.[1];
      if (otp) {
        document.getElementById(`active-otp-${orderId}`).style.display = 'flex';
        document.getElementById(`active-otp-code-${orderId}`).textContent = otp;
        document.getElementById(`active-waiting-${orderId}`).style.display = 'none';
        document.getElementById(`finish-btn-${orderId}`).style.display = 'inline-block';
        playBeep();
        // 高亮卡片
        document.getElementById(`active-card-${orderId}`)
                ?.classList.add('active-order-card--received');
      }
    }
  };

  ws.onclose = () => {
    delete activeOrderWSMap[orderId];
    // 若订单仍活跃，3 秒重连
    if (document.getElementById(`active-card-${orderId}`)) {
      setTimeout(() => connectActiveOrderWS(orderId), 3000);
    }
  };

  activeOrderWSMap[orderId] = ws;
}

function startActiveTimer(orderId, seconds) {
  let s = seconds;
  const el = document.getElementById(`timer-${orderId}`);
  if (!el) return;
  const iv = setInterval(() => {
    s--;
    if (!el) { clearInterval(iv); return; }
    el.textContent = formatTime(s);
    if (s <= 60)  el.style.color = '#ef4444';
    else if (s <= 300) el.style.color = '#f59e0b';
    if (s <= 0) {
      clearInterval(iv);
      el.textContent = '已超时';
      // 移除卡片
      setTimeout(() => {
        document.getElementById(`active-card-${orderId}`)?.remove();
      }, 3000);
    }
  }, 1000);
}

async function cancelActiveOrder(orderId) {
  if (!confirm('确定取消此订单？')) return;
  await fetch(`/api/sms/order/${orderId}/cancel`, { method: 'POST' });
  document.getElementById(`active-card-${orderId}`)?.remove();
  if (activeOrderWSMap[orderId]) {
    activeOrderWSMap[orderId].close();
    delete activeOrderWSMap[orderId];
  }
}

async function finishActiveOrder(orderId) {
  await fetch(`/api/sms/order/${orderId}/finish`, { method: 'POST' });
  document.getElementById(`active-card-${orderId}`)?.remove();
}

function copyActiveNumber(number, btn) {
  navigator.clipboard.writeText(number.replace(/\s/g,''));
  btn.textContent = '已复制 ✓';
  setTimeout(() => { btn.textContent = '复制'; }, 2000);
}

function copyActiveOTP(orderId) {
  const otp = document.getElementById(`active-otp-code-${orderId}`).textContent;
  navigator.clipboard.writeText(otp);
  showToast('验证码已复制', 'success');
}

function formatTime(s) {
  if (s <= 0) return '00:00';
  const m = Math.floor(s / 60);
  return `${String(m).padStart(2,'0')}:${String(s % 60).padStart(2,'0')}`;
}

document.addEventListener('DOMContentLoaded', loadActiveOrders);
```

```javascript
// static/js/sms-history.js

let currentPage   = 1;
let currentStatus = 'all';
const PAGE_SIZE   = 20;

async function loadHistory() {
  const from   = document.getElementById('date-from')?.value || '';
  const to     = document.getElementById('date-to')?.value   || '';
  const params = new URLSearchParams({
    page: currentPage, size: PAGE_SIZE,
    status: currentStatus !== 'all' ? currentStatus : '',
    from, to,
  });

  const resp = await fetch('/api/sms/orders/history?' + params);
  const data = await resp.json();

  renderHistoryTable(data.orders || []);
  renderPagination(data.total, currentPage, PAGE_SIZE);
}

function renderHistoryTable(orders) {
  const tbody = document.getElementById('history-tbody');

  if (orders.length === 0) {
    tbody.innerHTML = `<tr><td colspan="7" class="empty-cell">暂无订单记录</td></tr>`;
    return;
  }

  tbody.innerHTML = orders.map(o => `
    <tr>
      <td>
        <div class="history-service">
          <img src="/static/icons/sms/${o.product?.toLowerCase()}.png" width="20"
               onerror="this.style.display='none'">
          <span>${o.product || '—'}</span>
        </div>
      </td>
      <td class="mono">${o.phone || '—'}</td>
      <td>${o.country || '—'}</td>
      <td class="otp-cell">${o.code || '—'}</td>
      <td>¥${((o.price || 0) / 100).toFixed(2)}</td>
      <td><span class="status-pill status-pill--${(o.status||'').toLowerCase()}">${statusLabel(o.status)}</span></td>
      <td class="time-cell">${new Date(o.created_at).toLocaleString()}</td>
    </tr>
  `).join('');
}

function statusLabel(s) {
  return { RECEIVED: '✅ 已接收', CANCELED: '❌ 已取消', TIMEOUT: '⏱ 超时', PENDING: '⏳ 等待' }[s] || s;
}

function filterByStatus(status, btn) {
  currentStatus = status;
  currentPage   = 1;
  document.querySelectorAll('.filter-tab').forEach(t => t.classList.remove('filter-tab--active'));
  btn.classList.add('filter-tab--active');
  loadHistory();
}

function renderPagination(total, page, size) {
  const totalPages = Math.ceil(total / size);
  const pag = document.getElementById('pagination');
  if (totalPages <= 1) { pag.innerHTML = ''; return; }

  let html = '';
  if (page > 1)           html += `<button onclick="gotoPage(${page-1})">← 上一页</button>`;
  html += `<span>${page} / ${totalPages}</span>`;
  if (page < totalPages)  html += `<button onclick="gotoPage(${page+1})">下一页 →</button>`;
  pag.innerHTML = html;
}

function gotoPage(p) { currentPage = p; loadHistory(); }

function exportCSV() {
  const rows = document.querySelectorAll('.history-table tbody tr');
  let csv = '平台,号码,国家,验证码,费用,状态,时间\n';
  rows.forEach(row => {
    const cells = [...row.querySelectorAll('td')].map(td => `"${td.textContent.trim()}"`);
    csv += cells.join(',') + '\n';
  });
  const blob = new Blob(['\uFEFF' + csv], { type: 'text/csv;charset=utf-8;' });
  const url  = URL.createObjectURL(blob);
  const a    = document.createElement('a');
  a.href = url; a.download = `sms-history-${Date.now()}.csv`;
  a.click(); URL.revokeObjectURL(url);
}

document.addEventListener('DOMContentLoaded', loadHistory);
```

```javascript
// static/js/sms-prices.js

let allPrices = [];

async function loadPrices() {
  const resp = await fetch('/api/sms/prices');
  const data = await resp.json();
  allPrices  = data.prices || [];

  // 填充国家下拉
  const countries = [...new Set(allPrices.map(p => p.country))].sort();
  const sel = document.getElementById('country-filter');
  countries.forEach(c => {
    const opt = document.createElement('option');
    opt.value = c; opt.textContent = c;
    sel.appendChild(opt);
  });

  renderPricesTable(allPrices);
  document.getElementById('prices-update-time').textContent =
    `最后更新：${new Date().toLocaleTimeString()}`;
}

function renderPricesTable(prices) {
  const tbody = document.getElementById('prices-tbody');
  if (prices.length === 0) {
    tbody.innerHTML = '<tr><td colspan="7" class="empty-cell">暂无数据</td></tr>';
    return;
  }

  tbody.innerHTML = prices.map(p => {
    const rateClass = p.rate >= 80 ? 'high' : (p.rate >= 50 ? 'medium' : 'low');
    const stockCls  = p.count > 50 ? 'in' : (p.count > 0 ? 'low' : 'out');
    return `
      <tr>
        <td>
          <div class="price-service">
            <img src="/static/icons/sms/${p.product?.toLowerCase()}.png" width="20"
                 onerror="this.style.display='none'">
            <span>${p.product}</span>
          </div>
        </td>
        <td>${p.country}</td>
        <td>${p.operator}</td>
        <td class="price-val">¥${(p.cost / 100).toFixed(2)}</td>
        <td><span class="stock-badge stock-badge--${stockCls}">${p.count}</span></td>
        <td><span class="rate-${rateClass}">${p.rate}%</span></td>
        <td>
          <a href="/sms/buy?service=${p.product.toLowerCase()}" class="buy-link-btn">
            购买
          </a>
        </td>
      </tr>
    `;
  }).join('');
}

function filterPriceTable(query) {
  const q       = (query || document.getElementById('price-search').value).toLowerCase();
  const country = document.getElementById('country-filter').value;
  const filtered = allPrices.filter(p =>
    (!q || p.product.toLowerCase().includes(q)) &&
    (!country || p.country === country)
  );
  renderPricesTable(filtered);
}

document.addEventListener('DOMContentLoaded', () => {
  loadPrices();
  setInterval(loadPrices, 5 * 60 * 1000);  // 5 分钟自动刷新
});
```
