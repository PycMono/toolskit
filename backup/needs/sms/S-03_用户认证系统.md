## 3. JWT 工具

```go
// internal/auth/jwt.go
package auth

import (
    "errors"
    "os"
    "time"
    "github.com/golang-jwt/jwt/v5"
)

var jwtSecret = []byte(os.Getenv("JWT_SECRET"))

type Claims struct {
    UserID int64  `json:"uid"`
    Email  string `json:"email"`
    jwt.RegisteredClaims
}

func SignToken(userID int64, email string) (string, error) {
    claims := Claims{
        UserID: userID,
        Email:  email,
        RegisteredClaims: jwt.RegisteredClaims{
            ExpiresAt: jwt.NewNumericDate(time.Now().Add(24 * time.Hour)),
            IssuedAt:  jwt.NewNumericDate(time.Now()),
        },
    }
    return jwt.NewWithClaims(jwt.SigningMethodHS256, claims).SignedString(jwtSecret)
}

func SignRefreshToken(userID int64) (string, error) {
    claims := Claims{
        UserID: userID,
        RegisteredClaims: jwt.RegisteredClaims{
            ExpiresAt: jwt.NewNumericDate(time.Now().Add(30 * 24 * time.Hour)),
            IssuedAt:  jwt.NewNumericDate(time.Now()),
        },
    }
    return jwt.NewWithClaims(jwt.SigningMethodHS256, claims).SignedString(jwtSecret)
}

func ParseToken(tokenStr string) (*Claims, error) {
    token, err := jwt.ParseWithClaims(tokenStr, &Claims{},
        func(t *jwt.Token) (interface{}, error) { return jwtSecret, nil })
    if err != nil || !token.Valid {
        return nil, errors.New("invalid token")
    }
    return token.Claims.(*Claims), nil
}
```

## 4. JWT 中间件

```go
// internal/middleware/jwt.go
package middleware

import (
    "net/http"
    "strings"
    "devtoolbox/internal/auth"
    "github.com/gin-gonic/gin"
)

func JWTRequired() gin.HandlerFunc {
    return func(c *gin.Context) {
        // 优先从 Cookie 取，其次从 Authorization Header 取
        tokenStr, _ := c.Cookie("sms_token")
        if tokenStr == "" {
            header := c.GetHeader("Authorization")
            tokenStr = strings.TrimPrefix(header, "Bearer ")
        }

        if tokenStr == "" {
            if c.Request.Header.Get("Accept") == "application/json" {
                c.JSON(http.StatusUnauthorized, gin.H{"error": "unauthorized"})
            } else {
                c.Redirect(http.StatusFound, "/sms/login?redirect="+c.Request.URL.Path)
            }
            c.Abort()
            return
        }

        claims, err := auth.ParseToken(tokenStr)
        if err != nil {
            c.SetCookie("sms_token", "", -1, "/", "", false, true)
            c.Redirect(http.StatusFound, "/sms/login")
            c.Abort()
            return
        }

        c.Set("userID", claims.UserID)
        c.Set("email",  claims.Email)
        c.Next()
    }
}
```

## 5. 注册接口

```go
// internal/handler/sms_auth.go

// POST /api/sms/auth/register
func SMSRegister(c *gin.Context) {
    var req struct {
        Email    string `json:"email"    binding:"required,email"`
        Password string `json:"password" binding:"required,min=8"`
    }
    if err := c.ShouldBindJSON(&req); err != nil {
        c.JSON(400, gin.H{"error": "邮箱格式不正确或密码少于8位"})
        return
    }

    // 检查邮箱是否已注册
    exists, _ := db.SMSUserExistsByEmail(req.Email)
    if exists {
        c.JSON(409, gin.H{"error": "该邮箱已注册"})
        return
    }

    hash, err := bcrypt.GenerateFromPassword([]byte(req.Password), bcrypt.DefaultCost)
    if err != nil {
        c.JSON(500, gin.H{"error": "服务器错误"})
        return
    }

    user, err := db.CreateSMSUser(model.SMSUser{
        Email:        req.Email,
        PasswordHash: string(hash),
        DisplayName:  strings.Split(req.Email, "@")[0],
        APIKey:       generateAPIKey(),
    })
    if err != nil {
        c.JSON(500, gin.H{"error": "注册失败"})
        return
    }

    token, _ := auth.SignToken(user.ID, user.Email)
    refresh, _ := auth.SignRefreshToken(user.ID)

    c.SetCookie("sms_token",         token,   86400,         "/", "", false, true)
    c.SetCookie("sms_refresh_token", refresh, 86400*30, "/", "", false, true)
    c.JSON(200, gin.H{"token": token, "user": user})
}

// POST /api/sms/auth/login
func SMSLogin(c *gin.Context) {
    var req struct {
        Email    string `json:"email"    binding:"required,email"`
        Password string `json:"password" binding:"required"`
    }
    if err := c.ShouldBindJSON(&req); err != nil {
        c.JSON(400, gin.H{"error": "参数错误"})
        return
    }

    user, err := db.GetSMSUserByEmail(req.Email)
    if err != nil || user == nil {
        c.JSON(401, gin.H{"error": "邮箱或密码错误"})
        return
    }

    if err := bcrypt.CompareHashAndPassword(
        []byte(user.PasswordHash), []byte(req.Password)); err != nil {
        c.JSON(401, gin.H{"error": "邮箱或密码错误"})
        return
    }

    token, _   := auth.SignToken(user.ID, user.Email)
    refresh, _ := auth.SignRefreshToken(user.ID)

    c.SetCookie("sms_token",         token,   86400,    "/", "", false, true)
    c.SetCookie("sms_refresh_token", refresh, 86400*30, "/", "", false, true)
    c.JSON(200, gin.H{"token": token, "user": user})
}

// POST /api/sms/auth/logout
func SMSLogout(c *gin.Context) {
    c.SetCookie("sms_token",         "", -1, "/", "", false, true)
    c.SetCookie("sms_refresh_token", "", -1, "/", "", false, true)
    c.JSON(200, gin.H{"ok": true})
}

// POST /api/sms/auth/refresh
func SMSRefreshToken(c *gin.Context) {
    refresh, err := c.Cookie("sms_refresh_token")
    if err != nil {
        c.JSON(401, gin.H{"error": "no refresh token"})
        return
    }
    claims, err := auth.ParseToken(refresh)
    if err != nil {
        c.JSON(401, gin.H{"error": "invalid refresh token"})
        return
    }
    token, _ := auth.SignToken(claims.UserID, claims.Email)
    c.SetCookie("sms_token", token, 86400, "/", "", false, true)
    c.JSON(200, gin.H{"token": token})
}

// 生成 API Key
func generateAPIKey() string {
    b := make([]byte, 32)
    rand.Read(b)
    return "sk-" + hex.EncodeToString(b)
}
```

## 6. Google OAuth

```go
// internal/handler/sms_oauth.go

var googleOAuthConfig = &oauth2.Config{
    ClientID:     os.Getenv("GOOGLE_CLIENT_ID"),
    ClientSecret: os.Getenv("GOOGLE_CLIENT_SECRET"),
    RedirectURL:  os.Getenv("BASE_URL") + "/api/sms/auth/google/callback",
    Scopes:       []string{"openid", "profile", "email"},
    Endpoint:     google.Endpoint,
}

// GET /api/sms/auth/google
func SMSGoogleOAuth(c *gin.Context) {
    state := generateStateToken()
    c.SetCookie("oauth_state", state, 600, "/", "", false, true)
    c.Redirect(302, googleOAuthConfig.AuthCodeURL(state))
}

// GET /api/sms/auth/google/callback
func SMSGoogleCallback(c *gin.Context) {
    // 验证 state
    cookieState, _ := c.Cookie("oauth_state")
    if c.Query("state") != cookieState {
        c.Redirect(302, "/sms/login?error=invalid_state")
        return
    }

    code := c.Query("code")
    tok, err := googleOAuthConfig.Exchange(c.Request.Context(), code)
    if err != nil {
        c.Redirect(302, "/sms/login?error=oauth_failed")
        return
    }

    // 获取 Google 用户信息
    client   := googleOAuthConfig.Client(c.Request.Context(), tok)
    resp, _  := client.Get("https://www.googleapis.com/oauth2/v2/userinfo")
    defer resp.Body.Close()

    var info struct {
        ID      string `json:"id"`
        Email   string `json:"email"`
        Name    string `json:"name"`
        Picture string `json:"picture"`
    }
    json.NewDecoder(resp.Body).Decode(&info)

    // Upsert 用户
    user, err := db.UpsertSMSUserByGoogle(info.ID, info.Email, info.Name, info.Picture)
    if err != nil {
        c.Redirect(302, "/sms/login?error=db_error")
        return
    }

    token, _   := auth.SignToken(user.ID, user.Email)
    refresh, _ := auth.SignRefreshToken(user.ID)

    c.SetCookie("sms_token",         token,   86400,    "/", "", false, true)
    c.SetCookie("sms_refresh_token", refresh, 86400*30, "/", "", false, true)
    c.Redirect(302, "/sms/buy")
}
```

## 9. JavaScript — 登录/注册逻辑

```javascript
// static/js/sms-auth.js

async function submitLogin() {
  const email    = document.getElementById('login-email').value.trim();
  const password = document.getElementById('login-password').value;
  const btn      = document.getElementById('login-btn');
  const errEl    = document.getElementById('login-error');

  if (!email || !password) {
    showAuthError(errEl, '请填写邮箱和密码');
    return;
  }

  btn.disabled    = true;
  btn.textContent = '登录中...';
  errEl.style.display = 'none';

  try {
    const resp = await fetch('/api/sms/auth/login', {
      method:  'POST',
      headers: { 'Content-Type': 'application/json' },
      body:    JSON.stringify({ email, password }),
    });
    const data = await resp.json();

    if (!resp.ok) {
      showAuthError(errEl, data.error || '登录失败');
      return;
    }

    // 成功 → 跳转
    const params   = new URLSearchParams(location.search);
    const redirect = params.get('redirect') || '/sms/buy';
    window.location.href = redirect;
  } catch {
    showAuthError(errEl, '网络错误，请重试');
  } finally {
    btn.disabled    = false;
    btn.textContent = '登录';
  }
}

function showAuthError(el, msg) {
  el.textContent   = msg;
  el.style.display = 'block';
}

function togglePwd(inputId) {
  const el = document.getElementById(inputId);
  el.type  = el.type === 'password' ? 'text' : 'password';
}

// 按回车提交
document.addEventListener('keydown', (e) => {
  if (e.key === 'Enter') {
    const loginBtn    = document.getElementById('login-btn');
    const registerBtn = document.getElementById('register-btn');
    if (loginBtn)    submitLogin();
    if (registerBtn) submitRegister();
  }
});
```
