## 3. JavaScript

```javascript
// static/js/sms-account.js

let selectedAmount = 50;  // 默认选中 ¥50

function selectAmount(amount) {
  selectedAmount = amount;
  document.querySelectorAll('.amount-btn').forEach(btn => {
    btn.classList.toggle('amount-btn--active', parseInt(btn.dataset.amount) === amount);
  });
  document.getElementById('custom-amount').value = '';
}

function onCustomAmount(val) {
  selectedAmount = parseInt(val) || 0;
  document.querySelectorAll('.amount-btn').forEach(btn => btn.classList.remove('amount-btn--active'));
}

async function submitTopup() {
  if (!selectedAmount || selectedAmount < 1) {
    showToast('请选择或输入充值金额', 'error');
    return;
  }

  const method = document.querySelector('input[name="pay_method"]:checked')?.value;
  const btn    = document.getElementById('topup-btn');

  btn.disabled    = true;
  btn.textContent = '跳转支付...';

  try {
    const resp = await fetch('/api/sms/payment/create', {
      method:  'POST',
      headers: { 'Content-Type': 'application/json' },
      body:    JSON.stringify({ amount: selectedAmount * 100, method }),
    });
    const data = await resp.json();

    if (!resp.ok) throw new Error(data.error || '创建订单失败');

    // 跳转支付页面
    if (data.payUrl) window.location.href = data.payUrl;
    else if (data.qrCode) showQRCode(data.qrCode);

  } catch (err) {
    showToast(err.message, 'error');
  } finally {
    btn.disabled    = false;
    btn.textContent = '确认充值';
  }
}

async function copyAPIKey() {
  const key    = document.getElementById('apikey-text').textContent.trim();
  const btn    = document.getElementById('copy-apikey-btn');
  await navigator.clipboard.writeText(key);
  btn.textContent = '已复制 ✓';
  setTimeout(() => { btn.textContent = '复制'; }, 2000);
}

async function regenAPIKey() {
  if (!confirm('重新生成 API Key 后，旧的 Key 将立即失效。确定要继续吗？')) return;
  try {
    const resp = await fetch('/api/sms/account/regen-apikey', { method: 'POST' });
    const data = await resp.json();
    if (data.api_key) {
      document.getElementById('apikey-text').textContent = data.api_key;
      showToast('API Key 已更新', 'success');
    }
  } catch {
    showToast('重新生成失败，请重试', 'error');
  }
}

async function logout() {
  await fetch('/api/sms/auth/logout', { method: 'POST' });
  window.location.href = '/sms/login';
}

// 默认选中 ¥50
document.addEventListener('DOMContentLoaded', () => selectAmount(50));
```

## 4. 后端接口

```go
// GET /api/sms/account/me
func SMSGetMe(c *gin.Context) {
    userID := c.GetInt64("userID")
    user, err := db.GetSMSUserByID(userID)
    if err != nil { c.JSON(404, gin.H{"error": "user not found"}); return }
    c.JSON(200, user)
}

// POST /api/sms/account/regen-apikey
func SMSRegenAPIKey(c *gin.Context) {
    userID := c.GetInt64("userID")
    newKey := generateAPIKey()
    if err := db.UpdateSMSUserAPIKey(userID, newKey); err != nil {
        c.JSON(500, gin.H{"error": "failed"}); return
    }
    c.JSON(200, gin.H{"api_key": newKey})
}
```

## 5. 验收标准

- [ ] 账户头部显示头像（Google 头像 or 首字母）、邮箱、余额
- [ ] 余额格式 `¥XX.XX`，2 位小数
- [ ] 6 个充值金额快选按钮，点击后激活态高亮，¥50 标注 🔥 热门
- [ ] 自定义金额输入后，快选按钮取消激活态
- [ ] 支付方式三选一（支付宝/微信/USDT），单选高亮
- [ ] API Key 显示脱敏（可全选复制），「复制」2 秒后恢复
- [ ] 「重新生成」弹出 confirm 确认框，确认后更新页面显示
- [ ] 「退出登录」调用接口清除 Cookie 后跳转 `/sms/login`
