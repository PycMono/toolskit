## 4. JavaScript — 三级联动逻辑

```javascript
// static/js/sms-buy.js

let selectedService  = null;
let selectedCountry  = null;
let selectedOperator = null;
let allServices      = [];
let allCountries     = [];

/* ── Step 1: 加载服务列表 ──────────────────── */
async function loadServices() {
  try {
    const resp = await fetch('/api/sms/products');
    const data = await resp.json();
    allServices = data.products || [];
    renderServices(allServices);
  } catch (err) {
    showToast('加载服务失败，请刷新', 'error');
  }
}

function renderServices(services) {
  const grid    = document.getElementById('service-grid');
  const loading = document.getElementById('service-loading');
  if (loading) loading.remove();

  if (services.length === 0) {
    grid.innerHTML = '<p class="empty-hint">未找到相关平台</p>';
    return;
  }

  grid.innerHTML = services.map(s => `
    <div class="service-card ${s.name === selectedService?.name ? 'service-card--active' : ''}"
         onclick="selectService(${JSON.stringify(s).replace(/"/g, '&quot;')})">
      <img src="/static/icons/sms/${s.name.toLowerCase()}.png" alt="${s.displayName}"
           onerror="this.src='/static/icons/sms/default.png'">
      <span class="service-card__name">${s.displayName}</span>
    </div>
  `).join('');
}

function filterServices(query) {
  const filtered = query.trim()
    ? allServices.filter(s =>
        s.displayName.toLowerCase().includes(query.toLowerCase()) ||
        s.name.toLowerCase().includes(query.toLowerCase())
      )
    : allServices;
  renderServices(filtered);
}

function selectService(service) {
  selectedService  = service;
  selectedCountry  = null;
  selectedOperator = null;
  updateSummary();
  updateStepIndicator(2);
  loadCountries(service.name);
  showPanel('country');
}

/* ── Step 2: 加载国家列表 ──────────────────── */
async function loadCountries(serviceName) {
  document.getElementById('country-list').innerHTML =
    '<div class="loading-hint">加载中...</div>';
  try {
    const resp = await fetch(`/api/sms/countries?service=${serviceName}`);
    const data = await resp.json();
    allCountries = data.countries || [];
    renderCountries(allCountries);
  } catch (err) {
    showToast('加载国家失败', 'error');
  }
}

function renderCountries(countries) {
  const list = document.getElementById('country-list');
  if (countries.length === 0) {
    list.innerHTML = '<p class="empty-hint">暂无可用国家</p>';
    return;
  }
  list.innerHTML = countries.map(c => {
    const stockClass = c.qty > 100 ? 'high' : (c.qty > 0 ? 'normal' : 'none');
    return `
      <div class="country-item ${c.name === selectedCountry?.name ? 'country-item--active' : ''}"
           onclick='selectCountry(${JSON.stringify(c).replace(/"/g, "&quot;")})'>
        <span class="country-flag">${c.flag || '🌐'}</span>
        <span class="country-name">${c.displayName}</span>
        <span class="country-stock country-stock--${stockClass}">${c.qty} 个</span>
      </div>
    `;
  }).join('');
}

function filterCountries(query) {
  const filtered = query.trim()
    ? allCountries.filter(c =>
        c.displayName.toLowerCase().includes(query.toLowerCase())
      )
    : allCountries;
  renderCountries(filtered);
}

function selectCountry(country) {
  selectedCountry  = country;
  selectedOperator = null;
  updateSummary();
  updateStepIndicator(3);
  loadOperators(selectedService.name, country.name);
  showPanel('operator');
}

/* ── Step 3: 加载运营商列表 ────────────────── */
async function loadOperators(serviceName, countryName) {
  document.getElementById('operator-list').innerHTML =
    '<tr><td colspan="5" class="loading-hint">加载中...</td></tr>';
  try {
    const resp = await fetch(
      `/api/sms/operators?service=${serviceName}&country=${countryName}`
    );
    const data = await resp.json();
    renderOperators(data.operators || []);
  } catch (err) {
    showToast('加载运营商失败', 'error');
  }
}

function renderOperators(operators) {
  const tbody = document.getElementById('operator-list');
  if (operators.length === 0) {
    tbody.innerHTML = '<tr><td colspan="5" class="empty-hint">暂无可用运营商</td></tr>';
    return;
  }

  tbody.innerHTML = operators.map(op => {
    const rateClass = op.rate >= 80 ? 'high' : (op.rate >= 50 ? 'medium' : 'low');
    const stockClass = op.qty > 50 ? 'in' : (op.qty > 0 ? 'low' : 'out');
    const disabled  = op.qty === 0 ? 'disabled' : '';
    return `
      <tr>
        <td><span class="operator-name">${op.displayName || op.name}</span></td>
        <td><span class="operator-price">¥${(op.price / 100).toFixed(2)}</span></td>
        <td>
          <span class="stock-badge stock-badge--${stockClass}">
            ${op.qty > 0 ? op.qty + ' 个' : '无库存'}
          </span>
        </td>
        <td><span class="rate-${rateClass}">${op.rate}%</span></td>
        <td>
          <button class="select-operator-btn" ${disabled}
                  onclick='selectOperator(${JSON.stringify(op).replace(/"/g, "&quot;")})'>
            选择
          </button>
        </td>
      </tr>
    `;
  }).join('');
}

function selectOperator(operator) {
  selectedOperator = operator;
  updateSummary();
  // 不切换面板，停在第三步，右侧 Buy 按钮变为可用
}

/* ── 面板切换 ──────────────────────────────── */
function showPanel(name) {
  ['service', 'country', 'operator'].forEach(p => {
    document.getElementById(`panel-${p}`).style.display = p === name ? 'block' : 'none';
  });
}

function goBackToStep(step) {
  if (step === 1) {
    selectedService = null;
    selectedCountry = null;
    selectedOperator = null;
    updateStepIndicator(1);
    showPanel('service');
  } else if (step === 2) {
    selectedCountry = null;
    selectedOperator = null;
    updateStepIndicator(2);
    showPanel('country');
  }
  updateSummary();
}

/* ── 步骤指示器更新 ─────────────────────────── */
function updateStepIndicator(activeStep) {
  document.querySelectorAll('.step-indicator').forEach((el, idx) => {
    const step = idx + 1;
    el.classList.remove('step-indicator--active', 'step-indicator--done');
    if (step < activeStep)  el.classList.add('step-indicator--done');
    if (step === activeStep) el.classList.add('step-indicator--active');
  });
}

/* ── 汇总面板更新 ──────────────────────────── */
function updateSummary() {
  const show = (id, val) => {
    const row    = document.getElementById(`summary-${id}`);
    const valEl  = document.getElementById(`summary-${id}-val`);
    if (val && row && valEl) { row.style.display = 'flex'; valEl.textContent = val; }
    else if (row)              row.style.display = 'none';
  };

  show('service',  selectedService?.displayName);
  show('country',  selectedCountry ? `${selectedCountry.flag} ${selectedCountry.displayName}` : null);
  show('operator', selectedOperator?.displayName || selectedOperator?.name);

  const priceEl = document.getElementById('summary-price');
  if (priceEl) {
    priceEl.textContent = selectedOperator
      ? `¥${(selectedOperator.price / 100).toFixed(2)}`
      : '—';
  }

  const buyBtn = document.getElementById('buy-now-btn');
  if (buyBtn) {
    buyBtn.disabled = !selectedOperator;
  }
}
```
