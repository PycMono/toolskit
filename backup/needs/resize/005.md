function onDragOver(e) {

  }

function onDrop(e) {
  e.preventDefault(); e.stopPropagation();
  document.getElementById('uploadZone').classList.remove('ir-upload-zone--dragover');
  const files = e.dataTransfer?.files;
  if (files?.length > 0) addFiles(files);
}

function onFileSelect(input) {
  if (input.files?.length > 0) { addFiles(input.files); input.value = ''; }

/* ── 文件添加后 UI ─────────────────────────── */
async function onFilesAdded() {
  document.getElementById('optionsPanel').style.display = 'block';
  document.getElementById('optionsPanel')
    .scrollIntoView({ behavior: 'smooth', block: 'nearest' });

  // 读取第一张图片尺寸，填充提示和像素输入框默认值
  if (IRState.files.length > 0) {
    const dims = await loadFirstImageDimensions(IRState.files[0]);
    if (dims) {
      // 更新原始尺寸提示
      const hint = document.getElementById('origDimHint');
      if (hint) hint.textContent = `原始尺寸：${dims.w} × ${dims.h} px`;

      // 像素模式默认填入原始尺寸
      const wInput = document.getElementById('pixelWidth');
      const hInput = document.getElementById('pixelHeight');
      if (wInput && !wInput.value) wInput.value = dims.w;
      if (hInput && !hInput.value) hInput.value = dims.h;

      // 更新百分比预算
      updatePercentPreview(dims.w, dims.h,
        parseInt(document.getElementById('percentSlider')?.value || 100));
    }
  }
}

/* ── 模式切换 ──────────────────────────────── */
function switchMode(mode) {
  document.querySelectorAll('.ir-mode-tab').forEach(t => {
    t.classList.toggle('ir-mode-tab--active', t.dataset.mode === mode);

  document.getElementById('panelPixel').style.display   = mode === 'pixel'   ? 'block' : 'none';

  IRState.mode = mode;

/* ── 宽高输入联动（锁定比例）──────────────── */
function onWidthChange(val) {
  if (!IRState.lockRatio || IRState.files.length === 0) return;
  const w = parseInt(val);
  if (!w || w < 1) return;

  // 用第一张图计算比例
  const first = IRState.files[0];
  loadFirstImageDimensions(first).then(dims => {
    if (!dims) return;
    const newH = Math.round(dims.h * (w / dims.w));
    const hInput = document.getElementById('pixelHeight');
    if (hInput) hInput.value = newH;
  });
}

function onHeightChange(val) {
  if (!IRState.lockRatio || IRState.files.length === 0) return;
  const h = parseInt(val);
  if (!h || h < 1) return;

  const first = IRState.files[0];
  loadFirstImageDimensions(first).then(dims => {
    if (!dims) return;
    const newW = Math.round(dims.w * (h / dims.h));
    const wInput = document.getElementById('pixelWidth');
    if (wInput) wInput.value = newW;
  });
}

/* ── 宽高比锁定切换 ────────────────────────── */
function toggleLock() {
  IRState.lockRatio = !IRState.lockRatio;
  const btn  = document.getElementById('lockBtn');
  const icon = document.getElementById('lockIcon');
  if (!btn || !icon) return;

  if (IRState.lockRatio) {
    btn.classList.add('ir-lock-btn--locked');
    // 锁图标
    icon.innerHTML = `
      <rect x="3" y="11" width="18" height="11" rx="2"/>
      <path d="M7 11V7a5 5 0 0 1 10 0v4"/>
    `;
  } else {
    btn.classList.remove('ir-lock-btn--locked');
    // 解锁图标
    icon.innerHTML = `
      <rect x="3" y="11" width="18" height="11" rx="2"/>
      <path d="M7 11V7a5 5 0 0 1 9.9-1"/>
    `;
  }
}

/* ── 百分比滑块 ────────────────────────────── */
function onPercentChange(val) {
  document.getElementById('percentValue').textContent = val;
  const pct    = ((val - 10) / 190) * 100;
  const slider = document.getElementById('percentSlider');
  slider.style.setProperty('--slider-pct', `${pct}%`);

  // 更新预算尺寸
  if (IRState.files.length > 0) {
    loadFirstImageDimensions(IRState.files[0]).then(dims => {
      if (dims) updatePercentPreview(dims.w, dims.h, parseInt(val));
    });
  }
}

function updatePercentPreview(origW, origH, pct) {
  const el = document.getElementById('percentPreview');
  if (!el) return;
  const newW = Math.max(1, Math.round(origW * pct / 100));
  const newH = Math.max(1, Math.round(origH * pct / 100));
  el.innerHTML = `
    <span>原始：<strong>${origW} × ${origH}</strong></span>
    <span>→</span>
    <span>输出：<strong>${newW} × ${newH}</strong></span>
  `;
}

/* ══════════════════════════════════════════════
   结果卡片渲染 / 更新
════════════════════════════════════════════════ */
function upsertResultCard(data) {
  const list     = document.getElementById('resultsList');
  let   card     = document.getElementById(`ir-card-${data.id}`);
  const isNew    = !card;

  if (isNew) {
    card           = document.createElement('div');
    card.className = 'ir-result-card';
    card.id        = `ir-card-${data.id}`;
    card.style.opacity   = '0';
    card.style.transform = 'translateY(12px)';
    list.appendChild(card);
    requestAnimationFrame(() => {
      card.style.transition = 'opacity 0.3s ease, transform 0.3s ease';
      card.style.opacity    = '1';
      card.style.transform  = 'translateY(0)';
    });
  }

  // 原始文件信息（status = processing 时先显示原图缩略图）
  const origFile = IRState.files.find(f => f._irId === data.id);
  const thumbSrc = origFile ? URL.createObjectURL(origFile) : '';

  switch (data.status) {

    case 'processing':
      card.innerHTML = `
        <div class="ir-result-card__thumb">
          ${thumbSrc ? `<img src="${thumbSrc}" alt="${escapeHtml(data.name)}"
            onload="URL.revokeObjectURL(this.src)">` : ''}
        </div>
        <div class="ir-result-card__body">
          <p class="ir-result-card__name" title="${escapeHtml(data.name)}">${escapeHtml(data.name)}</p>
          <div class="ir-result-card__meta">
            <span class="ir-result-size">${formatFileSize(data.origSize)}</span>
          </div>
          <div class="ir-result-progress" id="ir-progress-${data.id}">
            <div class="ir-result-progress__bar"></div>
          </div>
        </div>
        <div class="ir-result-card__actions">
          <span class="ir-status-badge ir-status-badge--processing">
            <span class="ir-spinner"></span> 处理中
          </span>
        </div>
      `;
      animateProgress(card.querySelector('.ir-result-progress__bar'));
      break;

    case 'done': {
      const extBadge = getFormatBadge(data.name, data.origName);
      card.innerHTML = `
        <div class="ir-result-card__thumb">
          <img src="${data.previewUrl}" alt="${escapeHtml(data.name)}">
        </div>
        <div class="ir-result-card__body">
          <p class="ir-result-card__name" title="${escapeHtml(data.name)}">${escapeHtml(data.name)}</p>
          <div class="ir-result-card__meta">
            <span class="ir-result-dim">${data.origW}×${data.origH}</span>
            <span class="ir-result-arrow">→</span>
            <span class="ir-result-dim ir-result-dim--new">
              ${data.targetW}×${data.targetH} px ${extBadge}
            </span>
          </div>
          <div class="ir-result-card__sizes">
            <span>${formatFileSize(data.origSize)}</span>
            <span class="ir-result-arrow">→</span>
            <span class="${data.outputSize < data.origSize ? 'ir-size--smaller' : 'ir-size--larger'}">
              ${formatFileSize(data.outputSize)}
            </span>
          </div>
        </div>
        <div class="ir-result-card__actions">
          <button class="ir-btn-preview" onclick="openPreview('${data.id}')">
            <svg width="13" height="13" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2">
              <path d="M1 12s4-8 11-8 11 8 11 8-4 8-11 8-11-8-11-8z"/>
              <circle cx="12" cy="12" r="3"/>
            </svg>
            对比
          </button>
          <button class="ir-btn-download-one" onclick="downloadOne('${data.id}')">⬇ 下载</button>
          <button class="ir-btn-remove" onclick="removeItem('${data.id}')">✕</button>
        </div>
      `;
      break;
    }

    case 'error':
      card.innerHTML = `
        <div class="ir-result-card__thumb ir-result-card__thumb--error">⚠</div>
        <div class="ir-result-card__body">
          <p class="ir-result-card__name">${escapeHtml(data.name)}</p>
          <p class="ir-error-text">${escapeHtml(data.error || '处理失败')}</p>
        </div>
        <div class="ir-result-card__actions">
          <span class="ir-status-badge ir-status-badge--error">失败</span>
          <button class="ir-btn-retry" onclick="retryItem('${data.id}')">重试</button>
        </div>
      `;
      break;
  }
}

/* ── 进度条伪进度动画 ────────────────────────── */
function animateProgress(barEl) {
  if (!barEl) return;
  let pct = 0;
  barEl.style.width = '0%';
  const tick = () => {
    if (pct >= 85) return;
    pct = Math.min(85, pct + Math.random() * 15 + 3);
    barEl.style.transition = `width ${0.3 + Math.random() * 0.4}s ease`;
    barEl.style.width = pct + '%';
    setTimeout(tick, 300 + Math.random() * 400);
  };
  setTimeout(tick, 80);
}

/* ── 格式角标 ──────────────────────────────── */
function getFormatBadge(outputName, originalName) {
  if (!outputName || !originalName) return '';
  const outExt  = outputName.split('.').pop()?.toUpperCase();
  const origExt = originalName.split('.').pop()?.toUpperCase();
  if (outExt === origExt) return '';
  return `<span class="ir-fmt-badge">${outExt}</span>`;
}

/* ── 移除单项 ──────────────────────────────── */
function removeItem(id) {
  const idx  = IRState.files.findIndex(f => f._irId === id);
  if (idx > -1) IRState.files.splice(idx, 1);
  const ridx = IRState.results.findIndex(r => r.id === id);
  if (ridx > -1) {
    const r = IRState.results[ridx];
    if (r.previewUrl) URL.revokeObjectURL(r.previewUrl);
    if (r.origUrl)    URL.revokeObjectURL(r.origUrl);
    IRState.results.splice(ridx, 1);
  }
  const card = document.getElementById(`ir-card-${id}`);
  if (card) {
    card.style.transition = 'opacity 0.2s, transform 0.2s';
    card.style.opacity = '0'; card.style.transform = 'translateX(20px)';
    setTimeout(() => card.remove(), 220);
  }
  if (IRState.files.length === 0) clearAll();
  else updateSummaryStats();
}

/* ── 重试 ──────────────────────────────────── */
async function retryItem(id) {
  const file = IRState.files.find(f => f._irId === id);
  if (!file) return;
  upsertResultCard({ id, name: file.name, origSize: file.size, status: 'processing' });
  try {
    const result = await resizeOne(file);
    const idx = IRState.results.findIndex(r => r.id === id);
    if (idx > -1) IRState.results[idx] = result;
    else IRState.results.push(result);
    upsertResultCard(result);
  } catch (err) {
    upsertResultCard({ id, name: file.name, origSize: file.size, status: 'error', error: err.message });
  }
  updateSummaryStats();
}

function closePreview() {
  const overlay = document.getElementById('irPreviewOverlay');
  if (!overlay) return;
  document.removeEventListener('keydown', overlay._keyHandler);
  overlay.classList.remove('ir-preview-overlay--show');
  setTimeout(() => overlay.remove(), 250);
}

function bindSliderEvents(containerId, dividerId, wrapId) {
  const compare = document.getElementById(containerId);
  const divider = document.getElementById(dividerId);
  const wrap    = document.getElementById(wrapId);
  if (!compare || !divider || !wrap) return;

  let dragging = false;
  const setPos = (clientX) => {
    const rect = compare.getBoundingClientRect();
    let pct    = ((clientX - rect.left) / rect.width) * 100;
    pct        = Math.max(2, Math.min(98, pct));
    divider.style.left = pct + '%';
    wrap.style.width   = pct + '%';
  };

  divider.addEventListener('mousedown',  e => { dragging = true; e.preventDefault(); });
  document.addEventListener('mousemove', e => { if (dragging) setPos(e.clientX); });
  document.addEventListener('mouseup',   () => { dragging = false; });
  divider.addEventListener('touchstart', e => { dragging = true; e.preventDefault(); });
  document.addEventListener('touchmove', e => { if (dragging) setPos(e.touches[0].clientX); });
  document.addEventListener('touchend',  () => { dragging = false; });
  compare.addEventListener('click', e => {
    if (e.target === divider || divider.contains(e.target)) return;
    setPos(e.clientX);
  });
}

/* ══ 工具 ══════════════════════════════════════ */
function escapeHtml(str) {
  return String(str)
    .replace(/&/g, '&amp;').replace(/</g, '&lt;')
    .replace(/>/g, '&gt;').replace(/"/g, '&quot;');
}

## 3. 完整数据流

```
onFileSelect / onDrop
    ↓
addFiles(fileList)              ← 格式/大小/数量/去重校验
    ↓
IRState.files.push()
onFilesAdded()                  ← 显示选项面板，读取第一张图尺寸
    ↓
[用户设置参数，点击「开始调整大小」]
    ↓
startResize()
    ├─ validateResizeParams()   ← 校验输入合法性
    ├─ 并行（最多 4 并发）
    │   ├─ upsertResultCard({status:'processing'})
    │   ├─ resizeOne(file)      ← Canvas drawImage + toBlob
    │   └─ upsertResultCard({status:'done',...})
    └─ updateSummaryStats()

[用户点击「对比」]
    ↓
openPreview(id)                 ← Before/After 滑块弹窗
bindSliderEvents()              ← 鼠标/触摸拖拽

[用户点击「⬇ 下载」]
    ↓
downloadOne(id)

[用户点击「打包下载全部」]
    ↓
downloadAll()                   ← JSZip + FileSaver.saveAs()
```

## 4. 验收标准

### 功能正确性
- [ ] 像素模式锁定：输入宽 500px，高自动按比例计算并更新
- [ ] 像素模式解锁：宽高各自独立输入，下载后尺寸与设置完全一致（可拉伸变形）
- [ ] 百分比 50%：批量 3 张图，每张宽高均为原始的一半
- [ ] 百分比 200%：图片放大，可正常打开
- [ ] 预设 Instagram Post（1080×1080）：横图宽度 ≤1080，竖图高度 ≤1080，无裁剪
- [ ] PNG 转 WebP：下载文件 `.webp` 扩展名，浏览器可打开
- [ ] PNG 转 JPG：透明背景变白色背景，文件正常

### 结果 UI
- [ ] 卡片显示：缩略图 + 原始尺寸 + 调整后尺寸（蓝色）+ 文件大小变化
- [ ] 处理中：蓝色进度条动画（伪进度）
- [ ] 完成后进度条消失，显示「对比」+「下载」+「✕」
- [ ] 格式转换时蓝色 `WEBP`/`JPG` 角标显示
- [ ] 文件变小显示绿色文件大小，变大显示橙色

### 预览弹窗
- [ ] 点击「对比」弹窗打开，尺寸信息栏正确显示原始和调整后尺寸
- [ ] 滑块可拖拽：左侧原图，右侧调整后
- [ ] 触摸设备可正常拖拽
- [ ] ESC 键或点击遮罩关闭弹窗

### 批量下载
