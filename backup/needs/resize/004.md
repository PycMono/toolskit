# Block R-03 · 图片调整大小 — 纯前端调整引擎

> **文件**：`/static/js/img-resize-engine.js`  
> **预估工时**：3h  
> **核心原则**：文件全程不离开浏览器

---

## 1. 调整技术说明

| 模式 | 实现方式 |
|------|---------|
| 像素模式（锁定比例）| 按 W 或 H 计算，保持原始纵横比 |
| 像素模式（不锁定）| 直接使用输入的 W×H，可能拉伸 |
| 百分比模式 | `targetW = origW * pct/100`，自动保持比例 |
| 预设模式 | 固定 W×H，默认「适应模式」（fit，不拉伸）|
| 格式转换 | Canvas toBlob(targetMime, quality) |

## 2. 核心引擎代码（`/static/js/img-resize-engine.js`）

```javascript
'use strict';

/* ═══════════════════════════════════════════════
   全局状态
═══════════════════════════════════════════════ */
const IRState = {
  files:        [],         // 原始 File 对象
  results:      [],         // 处理结果
  isRunning:    false,
  mode:         'pixel',    // 'pixel' | 'percent' | 'preset'
  pixelW:       null,       // 像素模式目标宽
  pixelH:       null,       // 像素模式目标高
  lockRatio:    true,       // 宽高比锁定
  percent:      100,        // 百分比
  presetW:      null,       // 预设宽
  presetH:      null,       // 预设高
  outputFormat: 'original', // 'original' | 'image/jpeg' | 'image/png' | 'image/webp'
  quality:      90,         // 1~100
};

const IR_MAX_FILES   = 20;
const IR_MAX_SIZE_MB = 10;
const IR_CONCURRENCY = 4;

/* ═══════════════════════════════════════════════
   文件校验 & 入队
═══════════════════════════════════════════════ */
function addFiles(fileList) {
  const allowed = ['image/jpeg', 'image/jpg', 'image/png', 'image/webp'];
  let added = 0;

  for (const file of Array.from(fileList)) {
    if (IRState.files.length >= IR_MAX_FILES) {
      showToast(`最多同时处理 ${IR_MAX_FILES} 张图片`, 'error'); break;
    }
    if (!allowed.includes(file.type)) {
      showToast(`${file.name}：不支持的格式`, 'error'); continue;
    }
    if (file.size > IR_MAX_SIZE_MB * 1024 * 1024) {
      showToast(`${file.name} 超过 ${IR_MAX_SIZE_MB}MB 限制`, 'error'); continue;
    }
    const key = `${file.name}-${file.size}`;
    if (IRState.files.some(f => `${f.name}-${f.size}` === key)) continue;

    file._irId = `${Date.now()}-${Math.random().toString(36).slice(2, 8)}`;
    IRState.files.push(file);
    added++;
  }

  if (added > 0) onFilesAdded();
}

/* ═══════════════════════════════════════════════
   读取当前第一张图片的原始尺寸（用于 UI 提示）
═══════════════════════════════════════════════ */
function loadFirstImageDimensions(file) {
  return new Promise(resolve => {
    const img = new Image();
    const url = URL.createObjectURL(file);
    img.onload = () => {
      URL.revokeObjectURL(url);
      resolve({ w: img.naturalWidth, h: img.naturalHeight });
    };
    img.onerror = () => { URL.revokeObjectURL(url); resolve(null); };
    img.src = url;
  });
}

/* ═══════════════════════════════════════════════
   主调整流程
═══════════════════════════════════════════════ */
async function startResize() {
  if (IRState.files.length === 0 || IRState.isRunning) return;

  // 读取当前参数
  IRState.mode         = getCurrentMode();
  IRState.outputFormat = document.querySelector('input[name="outputFormat"]:checked')?.value || 'original';
  IRState.quality      = parseInt(document.getElementById('qualitySlider')?.value || 90);

  // 校验参数
  if (!validateResizeParams()) return;

  IRState.isRunning = true;
  const pending = IRState.files.filter(f => !IRState.results.find(r => r.id === f._irId));
  if (pending.length === 0) { IRState.isRunning = false; return; }

  setStartBtnState('running');
  showResultsSection();

  for (const file of pending) {
    upsertResultCard({
      id: file._irId, name: file.name, origSize: file.size, status: 'processing',
    });
  }

  await runConcurrent(pending, IR_CONCURRENCY, async (file) => {
    try {
      const result = await resizeOne(file);
      IRState.results.push(result);
      upsertResultCard(result);
    } catch (err) {
      upsertResultCard({
        id: file._irId, name: file.name, origSize: file.size,
        status: 'error', error: err.message,
      });
    }
  });

  IRState.isRunning = false;
  setStartBtnState('idle');
  updateSummaryStats();
}

/* ═══════════════════════════════════════════════
   参数校验
═══════════════════════════════════════════════ */
function validateResizeParams() {
  const mode = IRState.mode;
  if (mode === 'pixel') {
    const w = parseInt(document.getElementById('pixelWidth')?.value);
    const h = parseInt(document.getElementById('pixelHeight')?.value);
    if ((!w || w < 1 || w > 10000) && (!h || h < 1 || h > 10000)) {
      showToast('请输入有效的宽度或高度（1~10000px）', 'error'); return false;
    }
  }
  if (mode === 'percent') {
    const pct = parseInt(document.getElementById('percentSlider')?.value);
    if (!pct || pct < 1 || pct > 1000) {
      showToast('请输入有效的缩放比例', 'error'); return false;
    }
  }
  if (mode === 'preset') {
    if (!IRState.presetW || !IRState.presetH) {
      showToast('请选择一个预设尺寸', 'error'); return false;
    }
  }
  return true;
}

/* ═══════════════════════════════════════════════
   单张图片调整核心
═══════════════════════════════════════════════ */
async function resizeOne(file) {
  return new Promise((resolve, reject) => {
    const img = new Image();
    const url = URL.createObjectURL(file);

    img.onload = () => {
      URL.revokeObjectURL(url);
      const origW = img.naturalWidth;
      const origH = img.naturalHeight;

      // 计算目标尺寸
      let targetW, targetH;
      const mode = IRState.mode;

      if (mode === 'pixel') {
        const inputW = parseInt(document.getElementById('pixelWidth')?.value) || 0;
        const inputH = parseInt(document.getElementById('pixelHeight')?.value) || 0;

        if (IRState.lockRatio) {
          // 锁定比例：优先用宽度计算
          if (inputW > 0) {
            targetW = inputW;
            targetH = Math.round(origH * (inputW / origW));
          } else if (inputH > 0) {
            targetH = inputH;
            targetW = Math.round(origW * (inputH / origH));
          } else {
            targetW = origW; targetH = origH; // 兜底
          }
        } else {
          // 解锁比例：直接使用输入值（可能拉伸）
          targetW = inputW > 0 ? inputW : origW;
          targetH = inputH > 0 ? inputH : origH;
        }

      } else if (mode === 'percent') {
        const pct = parseInt(document.getElementById('percentSlider')?.value) || 100;
        targetW = Math.max(1, Math.round(origW * pct / 100));
        targetH = Math.max(1, Math.round(origH * pct / 100));

      } else if (mode === 'preset') {
        // 预设：fit 模式（保持比例，适应到预设框内，不裁剪）
        const pw = IRState.presetW;
        const ph = IRState.presetH;
        const scaleW = pw / origW;
        const scaleH = ph / origH;
        const scale  = Math.min(scaleW, scaleH); // fit = 取较小缩放比
        targetW = Math.round(origW * scale);
        targetH = Math.round(origH * scale);
      }

      // 绘制 Canvas
      const canvas  = document.createElement('canvas');
      canvas.width  = targetW;
      canvas.height = targetH;
      const ctx = canvas.getContext('2d');

      // JPG 不支持透明，填白底
      const targetMime = IRState.outputFormat === 'original' ? file.type : IRState.outputFormat;
      if (targetMime === 'image/jpeg') {
        ctx.fillStyle = '#ffffff';
        ctx.fillRect(0, 0, targetW, targetH);
      }

      ctx.drawImage(img, 0, 0, targetW, targetH);

      // 导出
      const quality = targetMime === 'image/png' ? undefined : IRState.quality / 100;
      canvas.toBlob(blob => {
        if (!blob) { reject(new Error('Canvas toBlob 失败')); return; }

        const ext     = mimeToExt(targetMime);
        const outName = replaceExt(file.name, ext);

        resolve({
          id:          file._irId,
          name:        outName,
          origName:    file.name,
          origSize:    file.size,
          origType:    file.type,
          origW,
          origH,
          targetW,
          targetH,
          outputBlob:  blob,
          outputSize:  blob.size,
          outputMime:  targetMime,
          status:      'done',
          previewUrl:  URL.createObjectURL(blob),
          origUrl:     URL.createObjectURL(file),
        });
      }, targetMime, quality);
    };

    img.onerror = () => {
      URL.revokeObjectURL(url);
      reject(new Error('图片加载失败，文件可能已损坏'));
    };
    img.src = url;
  });
}

/* ═══════════════════════════════════════════════
   并发控制
═══════════════════════════════════════════════ */
async function runConcurrent(items, concurrency, fn) {
  const queue = [...items];
  async function worker() {
    while (queue.length > 0) { const item = queue.shift(); if (item) await fn(item); }
  }
  const workers = [];
  for (let i = 0; i < Math.min(concurrency, items.length); i++) workers.push(worker());
  await Promise.all(workers);
}

/* ═══════════════════════════════════════════════
   单张下载
═══════════════════════════════════════════════ */
function downloadOne(id) {
  const result = IRState.results.find(r => r.id === id);
  if (!result?.outputBlob) return;
  saveAs(result.outputBlob, result.name);
}

/* ═══════════════════════════════════════════════
   批量 ZIP 下载
═══════════════════════════════════════════════ */
async function downloadAll() {
  const done = IRState.results.filter(r => r.status === 'done' && r.outputBlob);
  if (done.length === 0) { showToast('没有可下载的文件', 'info'); return; }

  const btn = document.getElementById('downloadAllBtn');
  if (btn) { btn.disabled = true; btn.textContent = '打包中...'; }

  try {
    const zip    = new JSZip();
    const folder = zip.folder('resized-images');
    for (const r of done) folder.file(r.name, r.outputBlob);

    const zipBlob = await zip.generateAsync({
      type: 'blob', compression: 'DEFLATE',
      compressionOptions: { level: 1 },
    });
    const ts = new Date().toISOString().slice(0, 10);
    saveAs(zipBlob, `resized-images-${ts}.zip`);
    showToast(`已打包 ${done.length} 张图片`, 'success');
  } catch (err) {
    showToast('打包失败：' + err.message, 'error');
  } finally {
    if (btn) { btn.disabled = false; btn.textContent = '⬇ 打包下载全部'; }
  }
}

/* ═══════════════════════════════════════════════
   清空
═══════════════════════════════════════════════ */
function clearAll() {
  for (const r of IRState.results) {
    if (r.previewUrl) URL.revokeObjectURL(r.previewUrl);
    if (r.origUrl)    URL.revokeObjectURL(r.origUrl);
  }
  IRState.files = []; IRState.results = [];
  document.getElementById('resultsList').innerHTML    = '';
  document.getElementById('resultsHeader').innerHTML  = '';
  document.getElementById('resultsSection').style.display = 'none';
  document.getElementById('optionsPanel').style.display   = 'none';
  document.getElementById('uploadIdle').style.display     = 'block';
  setStartBtnState('idle');
  showToast('已清空全部文件', 'info');
}

/* ═══════════════════════════════════════════════
   工具函数
═══════════════════════════════════════════════ */
function getCurrentMode() {
  return document.querySelector('.ir-mode-tab--active')?.dataset.mode || 'pixel';
}

function mimeToExt(mime) {
  return { 'image/jpeg': 'jpg', 'image/png': 'png', 'image/webp': 'webp' }[mime] || 'jpg';
}

function replaceExt(filename, newExt) {
  const i = filename.lastIndexOf('.');
  return (i > 0 ? filename.slice(0, i) : filename) + '.' + newExt;
}

function formatFileSize(bytes) {
  if (!bytes)               return '0 B';
  if (bytes < 1024)         return `${bytes} B`;
  if (bytes < 1024 * 1024)  return `${(bytes / 1024).toFixed(1)} KB`;
  return `${(bytes / (1024 * 1024)).toFixed(2)} MB`;
}

function setStartBtnState(state) {
  const btn  = document.getElementById('startBtn');
  const text = document.getElementById('startBtnText');
  if (!btn) return;
  btn.disabled     = state === 'running';
  text.textContent = state === 'running' ? '处理中...' : '开始调整大小';
}

function updateSummaryStats() {
  const done = IRState.results.filter(r => r.status === 'done');
  const el   = document.getElementById('resultsHeader');
  if (!el) return;
  el.innerHTML = `
    <div class="ir-stat">
      <span class="ir-stat__label">已处理</span>
      <span class="ir-stat__value">${done.length} 张</span>
    </div>
    ${done.length > 0 ? `
    <div class="ir-stat">
      <span class="ir-stat__label">输出尺寸</span>
      <span class="ir-stat__value ir-stat__value--blue">
        ${done[0].targetW} × ${done[0].targetH} px
      </span>
    </div>
    <div class="ir-stat">
      <span class="ir-stat__label">文件总大小</span>
      <span class="ir-stat__value">
        ${formatFileSize(done.reduce((s, r) => s + r.outputSize, 0))}
      </span>
    </div>` : ''}
  `;
}

function showToast(msg, type = 'info') {
  const container = document.getElementById('toastContainer');
  if (!container) return;
  const el = document.createElement('div');
  el.className   = `ir-toast ir-toast--${type}`;
  el.textContent = msg;
  container.appendChild(el);
  requestAnimationFrame(() => el.classList.add('ir-toast--show'));
  setTimeout(() => {
    el.classList.remove('ir-toast--show');
    setTimeout(() => el.remove(), 300);
  }, 3500);
}
```

## 3. 验收标准

- [ ] 像素模式（锁定比例）：输入宽 800，高自动计算正确
- [ ] 像素模式（解锁比例）：宽高各自独立，下载后尺寸与输入一致（可拉伸）
- [ ] 百分比 50%：下载图片宽高均为原始的一半
- [ ] 百分比 200%：图片放大到两倍，画质损失在预期范围内
- [ ] 预设「YouTube Thumbnail 1280×720」：横图适应宽度，竖图适应高度，不裁剪不拉伸
- [ ] 格式转换：PNG 转 JPG 后白色背景正常，扩展名正确
- [ ] 20 张图片批量处理，约 5 秒内完成，页面不卡死
- [ ] 清空后所有 ObjectURL 释放，内存无持续增长
- [ ] ZIP 包含全部 done 状态图片，文件名与设置一致
