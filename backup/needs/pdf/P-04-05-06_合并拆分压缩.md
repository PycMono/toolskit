```javascript
// 注入 PDFToolConfig
window.PDFToolConfig = {
  toolName: 'merge',
  maxFiles: 20,
  maxSizeMB: 50,

  getParams() {
    return {
      outputFilename: document.getElementById('output-filename')?.value || 'merged.pdf',
      order: uploadedFiles.map((f, i) => i),  // 当前顺序
    };
  },

  // 文件添加后更新缩略图预览
  onFilesChanged(files) {
    renderMergePreview(files);
  },
};

// 监听文件变化，更新合并预览
const origRenderFileList = window.renderFileList;
window.renderFileList = function() {
  origRenderFileList?.();
  renderMergePreview(uploadedFiles);
};

function renderMergePreview(files) {
  const el = document.getElementById('merge-preview');
  if (!el) return;

  if (files.length === 0) {
    el.innerHTML = '<div class="merge-preview__empty">上传文件后，在此预览缩略图并调整顺序</div>';
    return;
  }

  el.innerHTML = files.map((f, i) => `
    <div class="merge-thumb" draggable="true"
         ondragstart="mergeDragStart(${i})"
         ondragover="mergeDragOver(event)"
         ondrop="mergeDrop(event, ${i})"
         data-idx="${i}">
      <div class="merge-thumb__preview">
        <svg width="24" height="28" viewBox="0 0 24 28" fill="none">
          <rect width="24" height="28" rx="3" fill="#fee2e2"/>
          <path d="M4 8h16M4 12h12M4 16h10" stroke="#e53e3e" stroke-width="1.5"/>
        </svg>
      </div>
      <div class="merge-thumb__order">${i + 1}</div>
      <p class="merge-thumb__name">${f.name}</p>
      <p class="merge-thumb__size">${formatFileSize(f.size)}</p>
    </div>
  `).join('');
}

let mergeDragIdx = null;

function mergeDragStart(idx) { mergeDragIdx = idx; }
function mergeDragOver(e)    { e.preventDefault(); }
function mergeDrop(e, targetIdx) {
  e.preventDefault();
  if (mergeDragIdx === null || mergeDragIdx === targetIdx) return;
  const moved = uploadedFiles.splice(mergeDragIdx, 1)[0];
  uploadedFiles.splice(targetIdx, 0, moved);
  mergeDragIdx = null;
  renderFileList();
}
```

```go
// internal/pdfengine/merge.go
package pdfengine

import (
    "fmt"
    "os"
    "os/exec"
    "path/filepath"
    "strings"
)

// Merge 使用 pdfcpu 合并多个 PDF
func Merge(inputPaths []string, outputPath string) error {
    // pdfcpu merge output.pdf file1.pdf file2.pdf ...
    args := append([]string{"merge", outputPath}, inputPaths...)
    cmd  := exec.Command("pdfcpu", args...)
    if out, err := cmd.CombinedOutput(); err != nil {
        return fmt.Errorf("pdfcpu merge failed: %s", string(out))
    }
    return nil
}

// GetPageCount 获取 PDF 页数
func GetPageCount(inputPath string) (int, error) {
    cmd := exec.Command("pdfcpu", "info", inputPath)
    out, err := cmd.Output()
    if err != nil { return 0, err }
    // 解析 "Page count: 12" 行
    for _, line := range strings.Split(string(out), "\n") {
        if strings.Contains(line, "Page count:") {
            var n int
            fmt.Sscanf(strings.TrimPrefix(line, "Page count:"), " %d", &n)
            return n, nil
        }
    }
    return 0, nil
}
```

## 3. Go 后端（拆分处理）

```go
// internal/pdfengine/split.go

// SplitByRange 按页码范围拆分
// ranges: [[1,3],[5,5],[7,9]] → 3 个文件
func SplitByRange(inputPath string, ranges [][2]int, outputDir string) ([]string, error) {
    var outputs []string
    for i, r := range ranges {
        outPath := filepath.Join(outputDir, fmt.Sprintf("split_%03d.pdf", i+1))
        // pdfcpu trim -pages 1-3 input.pdf output.pdf
        pageRange := fmt.Sprintf("%d-%d", r[0], r[1])
        cmd := exec.Command("pdfcpu", "trim",
            "-pages", pageRange, inputPath, outPath)
        if out, err := cmd.CombinedOutput(); err != nil {
            return nil, fmt.Errorf("split failed: %s", string(out))
        }
        outputs = append(outputs, outPath)
    }
    return outputs, nil
}

// SplitByInterval 每 n 页拆分一次
func SplitByInterval(inputPath string, n int, outputDir string) ([]string, error) {
    // pdfcpu split -span n input.pdf outputDir/
    cmd := exec.Command("pdfcpu", "split",
        "-span", strconv.Itoa(n), inputPath, outputDir)
    if out, err := cmd.CombinedOutput(); err != nil {
        return nil, fmt.Errorf("split failed: %s", string(out))
    }
    // 列出生成文件
    return filepath.Glob(filepath.Join(outputDir, "*.pdf"))
}

// ParsePageRange 解析页码字符串 "1-3, 5, 7-9" → [[1,3],[5,5],[7,9]]
func ParsePageRange(s string) ([][2]int, error) {
    var result [][2]int
    for _, part := range strings.Split(s, ",") {
        part = strings.TrimSpace(part)
        if part == "" { continue }
        if strings.Contains(part, "-") {
            var a, b int
            if _, err := fmt.Sscanf(part, "%d-%d", &a, &b); err != nil {
                return nil, fmt.Errorf("invalid range: %s", part)
            }
            result = append(result, [2]int{a, b})
        } else {
            var a int
            if _, err := fmt.Sscanf(part, "%d", &a); err != nil {
                return nil, fmt.Errorf("invalid page: %s", part)
            }
            result = append(result, [2]int{a, a})
        }
    }
    return result, nil
}
```

```go
// internal/pdfengine/compress.go

// GhostscriptQuality 四档画质映射到 Ghostscript -dPDFSETTINGS
var gsQuality = map[string]string{
    "high":   "/printer",  // 300 DPI
    "medium": "/ebook",    // 150 DPI（推荐）
    "low":    "/default",
    "screen": "/screen",   // 72 DPI（最小）
}

// Compress 使用 Ghostscript 压缩 PDF
func Compress(inputPath, outputPath, quality string) error {
    pdfSetting, ok := gsQuality[quality]
    if !ok { pdfSetting = gsQuality["medium"] }

    cmd := exec.Command("gs",
        "-sDEVICE=pdfwrite",
        "-dCompatibilityLevel=1.4",
        fmt.Sprintf("-dPDFSETTINGS=%s", pdfSetting),
        "-dNOPAUSE",
        "-dQUIET",
        "-dBATCH",
        fmt.Sprintf("-sOutputFile=%s", outputPath),
        inputPath,
    )

    if out, err := cmd.CombinedOutput(); err != nil {
        return fmt.Errorf("ghostscript compress failed: %s", string(out))
    }
    return nil
}
```
