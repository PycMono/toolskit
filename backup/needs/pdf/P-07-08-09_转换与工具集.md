const currentTool = window.__PDF_TOOL_NAME__ || 'to-word';
const meta        = TOOL_META[currentTool] || TOOL_META['to-word'];

window.PDFToolConfig = {
  toolName:  meta.name,
  maxFiles:  meta.maxFiles,
  maxSizeMB: 100,

  getParams() {
    const params = { outputExt: meta.outputExt };

    // DPI（图片转换）
    const dpiBtn = document.querySelector('[data-dpi].config-option--active');
    if (dpiBtn) params.dpi = parseInt(dpiBtn.dataset.dpi);

    // JPG 画质

    if (jpgQ) params.quality = parseInt(jpgQ.value);

    // 页码范围

    if (pagesRange?.value) params.pagesRange = pagesRange.value;

    // 多页打包
    const packMode = document.querySelector('[data-pack].config-option--active');
    if (packMode) params.packMode = packMode.dataset.pack;

    return params;
  },
};

// internal/pdfengine/convert.go

// PDFToWord 使用 LibreOffice 将 PDF 转为 DOCX
func PDFToWord(inputPath, outputDir string) (string, error) {
    // LibreOffice: soffice --headless --convert-to docx input.pdf --outdir output/
    cmd := exec.Command("soffice",
        "--headless", "--convert-to", "docx",
        inputPath, "--outdir", outputDir,
    )
    if out, err := cmd.CombinedOutput(); err != nil {
        return "", fmt.Errorf("libreoffice convert failed: %s", string(out))
    }
    // 找到生成的 docx 文件
    base := strings.TrimSuffix(filepath.Base(inputPath), ".pdf")
    return filepath.Join(outputDir, base+".docx"), nil
}

// PDFToExcel LibreOffice 转 XLSX
func PDFToExcel(inputPath, outputDir string) (string, error) {
    cmd := exec.Command("soffice",
        "--headless", "--convert-to", "xlsx",
        inputPath, "--outdir", outputDir)
    if out, err := cmd.CombinedOutput(); err != nil {
        return "", fmt.Errorf("libreoffice convert failed: %s", string(out))
    }
    base := strings.TrimSuffix(filepath.Base(inputPath), ".pdf")
    return filepath.Join(outputDir, base+".xlsx"), nil
}

// PDFToImages 使用 ImageMagick 或 pdftoppm 将 PDF 转图片
func PDFToImages(inputPath, outputDir, format string, dpi, quality int) ([]string, error) {
    // pdftoppm -r 150 -jpeg -jpegopt quality=90 input.pdf output/page
    args := []string{
        "-r", strconv.Itoa(dpi),
        "-" + format,        // -jpeg 或 -png
    }
    if format == "jpeg" {
        args = append(args, "-jpegopt", fmt.Sprintf("quality=%d", quality))
    }
    args = append(args, inputPath, filepath.Join(outputDir, "page"))

    cmd := exec.Command("pdftoppm", args...)
    if out, err := cmd.CombinedOutput(); err != nil {
        return nil, fmt.Errorf("pdftoppm failed: %s", string(out))
    }
    ext := ".jpg"
    if format == "png" { ext = ".png" }
    return filepath.Glob(filepath.Join(outputDir, "page*"+ext))
}

// PDFToText 提取 PDF 文本
func PDFToText(inputPath, outputPath string) error {
    cmd := exec.Command("pdftotext", "-layout", inputPath, outputPath)
    if out, err := cmd.CombinedOutput(); err != nil {
        return fmt.Errorf("pdftotext failed: %s", string(out))
    }
    return nil
}

// internal/pdfengine/convert_to_pdf.go

// OfficeToPDF Word/Excel/PPT → PDF（LibreOffice）
func OfficeToPDF(inputPath, outputDir string) (string, error) {
    cmd := exec.Command("soffice",
        "--headless", "--convert-to", "pdf",
        inputPath, "--outdir", outputDir)
    if out, err := cmd.CombinedOutput(); err != nil {
        return "", fmt.Errorf("libreoffice failed: %s", string(out))
    }
    exts := []string{".doc", ".docx", ".xls", ".xlsx", ".ppt", ".pptx", ".txt"}
    base := filepath.Base(inputPath)
    for _, ext := range exts {
        base = strings.TrimSuffix(base, ext)
    }
    return filepath.Join(outputDir, base+".pdf"), nil
}

// ImagesToPDF 多张图片合并为 PDF（ImageMagick convert）
func ImagesToPDF(imagePaths []string, outputPath, pageSize, orientation string) error {
    // convert -compress jpeg -quality 90 img1.jpg img2.jpg output.pdf
    args := []string{"-compress", "jpeg", "-quality", "90"}

    if pageSize == "a4" {
        if orientation == "landscape" {
            args = append(args, "-page", "842x595")
        } else {
            args = append(args, "-page", "595x842")
        }
        args = append(args, "-gravity", "center", "-extent", "595x842")
    }

    args = append(args, imagePaths...)
    args = append(args, outputPath)

    cmd := exec.Command("convert", args...)
    if out, err := cmd.CombinedOutput(); err != nil {
        return fmt.Errorf("imagemagick failed: %s", string(out))
    }
    return nil
}

// TextToPDF 纯文本转 PDF
func TextToPDF(inputPath, outputPath string, params TextToPDFParams) error {
    // 使用 enscript + ps2pdf 或 pdfcpu text
    cmd := exec.Command("enscript",
        "--font=Helvetica12",
        "--output=-",
        inputPath,
    )
    psOut, err := cmd.Output()
    if err != nil {
        return fmt.Errorf("enscript failed: %w", err)
    }

    // ps → pdf
    ps2pdfCmd := exec.Command("ps2pdf", "-", outputPath)
    ps2pdfCmd.Stdin = bytes.NewReader(psOut)
    if out, err := ps2pdfCmd.CombinedOutput(); err != nil {
        return fmt.Errorf("ps2pdf failed: %s", string(out))
    }
    return nil
}

type TextToPDFParams struct {
    Font     string
    FontSize int
    PageSize string
}

// internal/pdfengine/tools.go

// Rotate PDF 旋转
func Rotate(inputPath, outputPath string, angle int, pageRange string) error {
    // pdfcpu rotate -pages all 90 input.pdf output.pdf
    pages := "all"
    if pageRange != "" { pages = pageRange }
    cmd := exec.Command("pdfcpu", "rotate",
        "-pages", pages,
        strconv.Itoa(angle),
        inputPath, outputPath)
    if out, err := cmd.CombinedOutput(); err != nil {
        return fmt.Errorf("rotate failed: %s", string(out))
    }
    return nil
}

// AddTextWatermark 文字水印
func AddTextWatermark(inputPath, outputPath, text string, opts WatermarkOpts) error {
    // pdfcpu watermark add 'sc:0.5, rot:45, opacity:0.5, color:#FF0000, font:Helvetica, points:40' input.pdf output.pdf
    desc := fmt.Sprintf("sc:1, rot:45, opacity:%.2f, color:#%s, font:Helvetica, points:%d",
        float64(opts.Opacity)/100, opts.Color, opts.FontSize)
    if opts.Position == "tile" { desc += ", onTop:true, tile:true" }

    cmd := exec.Command("pdfcpu", "watermark", "add", "-mode", "text",
        fmt.Sprintf("%s:%s", text, desc),
        inputPath, outputPath)
    if out, err := cmd.CombinedOutput(); err != nil {
        return fmt.Errorf("watermark failed: %s", string(out))
    }
    return nil
}

// Encrypt PDF 加密
func Encrypt(inputPath, outputPath, userPwd, ownerPwd string, perms []string) error {
    // pdfcpu encrypt -opw "owner" -upw "user" -perm print:false input.pdf output.pdf
    args := []string{"encrypt"}
    if userPwd  != "" { args = append(args, "-upw", userPwd)  }
    if ownerPwd != "" { args = append(args, "-opw", ownerPwd) }
    for _, p := range perms { args = append(args, "-perm", p+":false") }
    args = append(args, inputPath, outputPath)

    cmd := exec.Command("pdfcpu", args...)
    if out, err := cmd.CombinedOutput(); err != nil {
        return fmt.Errorf("encrypt failed: %s", string(out))
    }
    return nil
}

// Decrypt PDF 解密
func Decrypt(inputPath, outputPath, password string) error {
    cmd := exec.Command("pdfcpu", "decrypt",
        "-upw", password, inputPath, outputPath)
    if out, err := cmd.CombinedOutput(); err != nil {
        if strings.Contains(string(out), "password") {
            return fmt.Errorf("wrong_password")
        }
        return fmt.Errorf("decrypt failed: %s", string(out))
    }
    return nil
}

// OCR 文字识别（Tesseract）
func OCR(inputPath, outputPath, lang string) error {
    // 先将 PDF 页转图片，再 OCR，最后合并为可搜索 PDF
    tmpDir, _ := os.MkdirTemp("", "ocr-*")
    defer os.RemoveAll(tmpDir)

    // step1: PDF → PNG
    cmd := exec.Command("pdftoppm", "-png", "-r", "200", inputPath,
        filepath.Join(tmpDir, "page"))
    if out, err := cmd.CombinedOutput(); err != nil {
        return fmt.Errorf("pdftoppm failed: %s", string(out))
    }

    // step2: Tesseract OCR → PDF
    pages, _ := filepath.Glob(filepath.Join(tmpDir, "page-*.png"))
    var ocrPDFs []string
    for i, pg := range pages {
        ocrOut := filepath.Join(tmpDir, fmt.Sprintf("ocr_%03d", i))
        cmd := exec.Command("tesseract", pg, ocrOut,
            "-l", lang, "pdf")
        if out, err := cmd.CombinedOutput(); err != nil {
            return fmt.Errorf("tesseract failed: %s", string(out))
        }
        ocrPDFs = append(ocrPDFs, ocrOut+".pdf")
    }

    // step3: 合并 OCR PDF
    return Merge(ocrPDFs, outputPath)
}

type WatermarkOpts struct {
    FontSize int
    Color    string  // hex，不含 #
    Opacity  int     // 0-100
    Position string  // center/tile/diagonal
}
