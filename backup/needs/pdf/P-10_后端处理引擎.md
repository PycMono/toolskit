# Block P-10 · PDF 处理引擎（后端基础设施）

> **所属模块**：PDF 工具后端核心  
> **预估工时**：5h  
> **依赖**：无（所有工具 Block 依赖此层）  
> **交付粒度**：文件上传API + 任务队列 + 工具链调度 + 轮询接口 + 下载接口 + 临时文件自动清理

---

## 1. 架构设计

```
前端
  ↓ POST /api/pdf/upload（可多次）→ fileId[]
  ↓ POST /api/pdf/process { tool, fileIds, params } → taskId
  ↓ GET  /api/pdf/task/:taskId（轮询） → {status, progress, ...}
  ↓ GET  /api/pdf/download/:taskId → 文件下载

后端任务队列
  ├─ worker goroutine（并发 3）
  ├─ 任务状态：pending → processing → done/error
  └─ 自动清理（1 小时后删除所有临时文件）
```

```go
package model

import "time"

type PDFTaskStatus string

const (
    TaskPending    PDFTaskStatus = "pending"

    TaskDone       PDFTaskStatus = "done"
    TaskError      PDFTaskStatus = "error"

type PDFTask struct {
    TaskID         string        `json:"taskId"`
    Tool           string        `json:"tool"`
    FileIDs        []string      `json:"fileIds"`
    Params         PDFTaskParams `json:"params"`
    Status         PDFTaskStatus `json:"status"`
    Progress       int           `json:"progress"`   // 0-100
    Message        string        `json:"message"`
    Error          string        `json:"error,omitempty"`
    OutputFileName string        `json:"outputFileName"`
    InputSize      int64         `json:"inputSize"`
    OutputSize     int64         `json:"outputSize"`
    Pages          int           `json:"pages"`
    CreatedAt      time.Time     `json:"createdAt"`
    ExpiresAt      time.Time     `json:"expiresAt"`  // 1小时后清理
}

type PDFTaskParams struct {
    // 通用
    OutputFilename string `json:"outputFilename,omitempty"`
    PagesRange     string `json:"pagesRange,omitempty"`

    // 拆分
    SplitMode     string `json:"splitMode,omitempty"`    // range/interval/extract
    SplitRange    string `json:"range,omitempty"`
    SplitInterval int    `json:"interval,omitempty"`
    ExtractPages  string `json:"extract,omitempty"`

    // 压缩
    Quality string `json:"quality,omitempty"`  // high/medium/low/screen

    // 旋转
    RotateAngle int    `json:"angle,omitempty"`
    RotatePages string `json:"rotatePages,omitempty"`

    // 水印
    WatermarkType     string `json:"watermarkType,omitempty"` // text/image
    WatermarkText     string `json:"watermarkText,omitempty"`
    WatermarkFontSize int    `json:"fontSize,omitempty"`
    WatermarkColor    string `json:"color,omitempty"`
    WatermarkOpacity  int    `json:"opacity,omitempty"`
    WatermarkPosition string `json:"position,omitempty"`
    WatermarkImageID  string `json:"watermarkImageId,omitempty"`

    Permissions   []string `json:"permissions,omitempty"`
```

## 4. 上传接口

```go
// internal/handler/pdf_upload.go
package handler

import (
    "fmt"
    "io"
    "net/http"
    "os"
    "path/filepath"
    "strings"

    "github.com/gin-gonic/gin"
    "github.com/google/uuid"
)

const (
    MaxUploadSize = 50 << 20      // 50 MB
    UploadBaseDir = "tmp/pdf/uploads"
)

// POST /api/pdf/upload
func PDFUpload(c *gin.Context) {
    c.Request.Body = http.MaxBytesReader(c.Writer, c.Request.Body, MaxUploadSize)

    file, header, err := c.Request.FormFile("file")
    if err != nil {
        if strings.Contains(err.Error(), "too large") {
            c.JSON(413, gin.H{"error": "文件过大，最大支持 50MB"})
        } else {
            c.JSON(400, gin.H{"error": "文件上传失败"})
        }
        return
    }
    defer file.Close()

    // 验证文件类型（按扩展名白名单）
    ext := strings.ToLower(filepath.Ext(header.Filename))
    allowed := map[string]bool{
        ".pdf": true, ".doc": true, ".docx": true,
        ".xls": true, ".xlsx": true, ".ppt": true, ".pptx": true,
        ".jpg": true, ".jpeg": true, ".png": true, ".txt": true,
    }
    if !allowed[ext] {
        c.JSON(400, gin.H{"error": "不支持的文件格式: " + ext})
        return
    }

    // 生成唯一 fileId，创建目录
    fileId  := uuid.New().String()
    dir     := filepath.Join(UploadBaseDir, fileId)
    if err := os.MkdirAll(dir, 0755); err != nil {
        c.JSON(500, gin.H{"error": "服务器错误"}); return
    }

    // 保存文件（保留原始扩展名，文件名固定为 original.ext）
    savePath := filepath.Join(dir, "original"+ext)
    out, err := os.Create(savePath)
    if err != nil { c.JSON(500, gin.H{"error": "保存文件失败"}); return }
    defer out.Close()

    size, _ := io.Copy(out, file)

    c.JSON(200, gin.H{
        "fileId":   fileId,
        "filename": header.Filename,
        "size":     size,
        "ext":      ext,
    })
}
```

```go
// POST /api/pdf/process
func PDFProcess(c *gin.Context) {
    var req struct {
        Tool    string               `json:"tool"    binding:"required"`
        FileIDs []string             `json:"fileIds" binding:"required,min=1"`
        Params  model.PDFTaskParams  `json:"params"`
    }
    if err := c.ShouldBindJSON(&req); err != nil {
        c.JSON(400, gin.H{"error": "参数错误: " + err.Error()}); return
    }

    // 验证所有 fileId 存在
    for _, fid := range req.FileIDs {
        dir := filepath.Join(UploadBaseDir, fid)
        if _, err := os.Stat(dir); os.IsNotExist(err) {
            c.JSON(400, gin.H{"error": "文件不存在或已过期: " + fid}); return
        }
    }

    task := &model.PDFTask{
        TaskID:    uuid.New().String(),
        Tool:      req.Tool,
        FileIDs:   req.FileIDs,
        Params:    req.Params,
        Status:    model.TaskPending,
        CreatedAt: time.Now(),
        ExpiresAt: time.Now().Add(time.Hour),
    }

    // 提交到任务队列
    pdfengine.Submit(task)

    c.JSON(200, gin.H{"taskId": task.TaskID})
}
```

```go
func PDFTaskStatus(c *gin.Context) {
    taskId := c.Param("taskId")
    task   := pdfengine.GetTask(taskId)
    if task == nil {
        c.JSON(404, gin.H{"error": "任务不存在"}); return
    }

    c.JSON(200, gin.H{
        "taskId":         task.TaskID,
        "status":         task.Status,
        "progress":       task.Progress,
        "message":        task.Message,
        "error":          task.Error,
        "outputFileName": task.OutputFileName,
        "inputSize":      task.InputSize,
        "outputSize":     task.OutputSize,
        "pages":          task.Pages,
    })
}

// GET /api/pdf/download/:taskId
func PDFDownload(c *gin.Context) {
    taskId := c.Param("taskId")
    task   := pdfengine.GetTask(taskId)
    if task == nil || task.Status != model.TaskDone {
        c.JSON(404, gin.H{"error": "文件不存在或尚未完成"}); return
    }

    outPath := filepath.Join("tmp/pdf/tasks", taskId, task.OutputFileName)
    if _, err := os.Stat(outPath); os.IsNotExist(err) {
        c.JSON(404, gin.H{"error": "文件已过期"}); return
    }

    c.Header("Content-Disposition",
        fmt.Sprintf(`attachment; filename="%s"`, task.OutputFileName))
    c.File(outPath)
}

// DELETE /api/pdf/file/:fileId
func PDFDeleteFile(c *gin.Context) {
    fileId := c.Param("fileId")
    // 安全检查：仅允许删除 uploads 目录下的文件
    dir := filepath.Join(UploadBaseDir, fileId)
    if !strings.HasPrefix(filepath.Clean(dir), filepath.Clean(UploadBaseDir)) {
        c.JSON(403, gin.H{"error": "forbidden"}); return
    }
    os.RemoveAll(dir)
    c.JSON(200, gin.H{"ok": true})
}
```

## 7. 任务引擎（Worker 池）

```go
// internal/pdfengine/engine.go
package pdfengine

import (
    "log"
    "sync"
    "time"
    "devtoolbox/internal/model"
)

const WorkerCount = 3  // 并发处理任务数

var (
    taskQueue = make(chan *model.PDFTask, 50)
    taskStore = sync.Map{}
)

func init() {
    // 启动 Worker 池
    for i := 0; i < WorkerCount; i++ {
        go worker()
    }
    // 启动清理任务
    go cleanupLoop()
}

func Submit(task *model.PDFTask) {
    taskStore.Store(task.TaskID, task)
    taskQueue <- task
}

func GetTask(taskId string) *model.PDFTask {
    if v, ok := taskStore.Load(taskId); ok {
        return v.(*model.PDFTask)
    }
    return nil
}

func worker() {
    for task := range taskQueue {
        processTask(task)
    }
}

func processTask(task *model.PDFTask) {
    task.Status   = model.TaskProcessing
    task.Progress = 10
    task.Message  = "处理中..."

    var err error

    switch task.Tool {
    case "merge":
        err = runMerge(task)
    case "split":
        err = runSplit(task)
    case "compress":
        err = runCompress(task)
    case "rotate":
        err = runRotate(task)
    case "watermark":
        err = runWatermark(task)
    case "encrypt":
        err = runEncrypt(task)
    case "decrypt":
        err = runDecrypt(task)
    case "ocr":
        err = runOCR(task)
    case "to-word", "to-excel", "to-ppt", "to-txt", "to-html":
        err = runConvertFromPDF(task)
    case "to-jpg", "to-png":
        err = runConvertToImages(task)
    case "from-word", "from-excel", "from-ppt", "from-txt":
        err = runConvertToPDF(task)
    case "from-jpg", "from-png":
        err = runImagesToPDF(task)
    default:
        err = fmt.Errorf("unknown tool: %s", task.Tool)
    }

    if err != nil {
        task.Status  = model.TaskError
        task.Error   = err.Error()
        task.Message = "处理失败"
        log.Printf("PDF task %s failed: %v", task.TaskID, err)
        return
    }

    task.Status   = model.TaskDone
    task.Progress = 100
    task.Message  = "处理完成"
}
```

```go
// internal/pdfengine/worker.go

func getInputPaths(fileIDs []string) ([]string, error) {
    var paths []string
    for _, fid := range fileIDs {
        // 在上传目录中找 original.* 文件
        dir   := filepath.Join(UploadBaseDir, fid)
        files, err := filepath.Glob(filepath.Join(dir, "original.*"))
        if err != nil || len(files) == 0 {
            return nil, fmt.Errorf("file not found: %s", fid)
        }
        paths = append(paths, files[0])
    }
    return paths, nil
}

func makeOutputDir(taskID string) (string, error) {
    dir := filepath.Join("tmp/pdf/tasks", taskID)
    return dir, os.MkdirAll(dir, 0755)
}

func getInputSize(paths []string) int64 {
    var total int64
    for _, p := range paths {
        if info, err := os.Stat(p); err == nil {
            total += info.Size()
        }
    }
    return total
}

// runMerge 合并任务执行
func runMerge(task *model.PDFTask) error {
    inputs, err := getInputPaths(task.FileIDs)
    if err != nil { return err }

    outDir, err := makeOutputDir(task.TaskID)
    if err != nil { return err }

    outName := task.Params.OutputFilename
    if outName == "" { outName = "merged.pdf" }

    outPath := filepath.Join(outDir, outName)

    task.InputSize = getInputSize(inputs)
    task.Progress  = 30

    if err := Merge(inputs, outPath); err != nil { return err }

    info, _ := os.Stat(outPath)
    if info != nil { task.OutputSize = info.Size() }

    pages, _ := GetPageCount(outPath)
    task.Pages          = pages
    task.OutputFileName = outName
    return nil
}

// runCompress 压缩任务执行
func runCompress(task *model.PDFTask) error {
    inputs, err := getInputPaths(task.FileIDs)
    if err != nil || len(inputs) == 0 { return fmt.Errorf("no input file") }

    outDir, err := makeOutputDir(task.TaskID)
    if err != nil { return err }

    outPath := filepath.Join(outDir, "compressed.pdf")
    task.InputSize = getInputSize(inputs)
    task.Progress  = 30

    quality := task.Params.Quality
    if quality == "" { quality = "medium" }

    if err := Compress(inputs[0], outPath, quality); err != nil { return err }

    info, _ := os.Stat(outPath)
    if info != nil { task.OutputSize = info.Size() }
    task.OutputFileName = "compressed.pdf"
    return nil
}

// runConvertToImages PDF → JPG/PNG，多页打包 ZIP
func runConvertToImages(task *model.PDFTask) error {
    inputs, err := getInputPaths(task.FileIDs)
    if err != nil || len(inputs) == 0 { return fmt.Errorf("no input") }

    outDir, err := makeOutputDir(task.TaskID)
    if err != nil { return err }

    dpi     := task.Params.DPI
    if dpi == 0 { dpi = 150 }
    quality := task.Params.JpgQuality
    if quality == 0 { quality = 90 }

    format := "jpeg"
    if task.Tool == "to-png" { format = "png" }

    task.Progress = 30
    images, err := PDFToImages(inputs[0], outDir, format, dpi, quality)
    if err != nil { return err }

    // 打包 ZIP
    task.Progress = 80
    zipName   := "pages.zip"
    zipPath   := filepath.Join(outDir, zipName)
    if err := zipFiles(images, zipPath); err != nil { return err }

    info, _ := os.Stat(zipPath)
    if info != nil { task.OutputSize = info.Size() }
    task.Pages          = len(images)
    task.OutputFileName = zipName
    return nil
}

// zipFiles 将文件列表打包为 ZIP
func zipFiles(files []string, zipPath string) error {
    zf, err := os.Create(zipPath)
    if err != nil { return err }
    defer zf.Close()

    w := zip.NewWriter(zf)
    defer w.Close()

    for _, f := range files {
        src, err := os.Open(f)
        if err != nil { continue }
        defer src.Close()

        dst, err := w.Create(filepath.Base(f))
        if err != nil { continue }
        io.Copy(dst, src)
    }
    return nil
}
```

## 9. 自动清理

```go
// internal/pdfengine/cleanup.go

func cleanupLoop() {
    ticker := time.NewTicker(10 * time.Minute)
    for range ticker.C {
        cleanupExpiredFiles()
    }
}

func cleanupExpiredFiles() {
    now := time.Now()

    // 清理任务目录
    taskBase := "tmp/pdf/tasks"
    entries, _ := os.ReadDir(taskBase)
    for _, entry := range entries {
        if !entry.IsDir() { continue }
        dir   := filepath.Join(taskBase, entry.Name())
        info, _ := entry.Info()
        // 超过 1 小时的任务目录清理
        if now.Sub(info.ModTime()) > time.Hour {
            os.RemoveAll(dir)
            taskStore.Delete(entry.Name())
        }
    }

    // 清理上传目录（2 小时未处理的文件）
    uploadBase := "tmp/pdf/uploads"
    uEntries, _ := os.ReadDir(uploadBase)
    for _, entry := range uEntries {
        if !entry.IsDir() { continue }
        dir  := filepath.Join(uploadBase, entry.Name())
        info, _ := entry.Info()
        if now.Sub(info.ModTime()) > 2*time.Hour {
            os.RemoveAll(dir)
        }
    }

    log.Printf("PDF cleanup done: tasks=%d uploads=%d",
        len(entries), len(uEntries))
}
```

## 12. 验收标准

- [ ] `POST /api/pdf/upload` 上传 PDF，返回 `{fileId, filename, size}`
- [ ] 超过 50MB 文件返回 413
- [ ] 不在白名单的文件类型返回 400
- [ ] `POST /api/pdf/process` 返回 taskId，任务状态初始为 `pending`
- [ ] `GET /api/pdf/task/:taskId` 轮询返回 `{status, progress, message}`
- [ ] 任务完成后 status 变 `done`，progress=100
- [ ] `GET /api/pdf/download/:taskId` 返回正确文件，Content-Disposition 头正确
- [ ] 3 个 Worker 并行处理，不互相阻塞
- [ ] 任务目录在 1 小时后被清理，不留存用户文件
- [ ] 压缩任务：输入 1MB PDF，medium 档输出明显小于输入
- [ ] 合并任务：3 个 PDF 合并为 1 个，页数之和正确
- [ ] 格式转换：PDF → DOCX，LibreOffice 正常工作
- [ ] OCR 任务：扫描版 PDF → 可搜索 PDF，tesseract 识别中英文
