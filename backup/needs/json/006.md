## 2. Tree Viewer JS（`/static/js/json-tool-tree.js`）

```javascript
function processJson() {
  const parsed = parseInput();
  if (parsed === null) return;

  const container = document.getElementById('treeOutput');
  container.innerHTML = '';
  container.appendChild(renderNode(parsed, 'root', '$'));
}

function renderNode(value, key, path) {
  const wrap = document.createElement('div');
  wrap.className = 'jt-tree-node';

  if (value === null || typeof value !== 'object') {
    // 叶子节点
    const leaf = document.createElement('div');
    leaf.className = 'jt-tree-leaf';
    leaf.innerHTML = `
      <span class="jt-tree-key" onclick="copyPath('${escapeHtml(path)}')"
            title="复制路径：${escapeHtml(path)}">${escapeHtml(String(key))}</span>
      <span class="jt-tree-sep">: </span>
      <span class="jt-tree-val jt-tree-val--${typeof value}">${renderValue(value)}</span>
    `;
    wrap.appendChild(leaf);
    return wrap;
  }

  const isArray   = Array.isArray(value);
  const childKeys = Object.keys(value);
  const count     = childKeys.length;
  const preview   = isArray ? `[${count} items]` : `{${count} keys}`;

  // 折叠区
  const header = document.createElement('div');
  header.className = 'jt-tree-header';
  header.innerHTML = `
    <button class="jt-tree-toggle" onclick="toggleNode(this)">▶</button>
    <span class="jt-tree-key" onclick="copyPath('${escapeHtml(path)}')"
          title="复制路径">${escapeHtml(String(key))}</span>
    <span class="jt-tree-sep">: </span>
    <span class="jt-tree-bracket">${isArray ? '[' : '{'}</span>
    <span class="jt-tree-preview">${preview}</span>
    <span class="jt-tree-bracket jt-tree-bracket--close">${isArray ? ']' : '}'}</span>
  `;

  const children = document.createElement('div');
  children.className = 'jt-tree-children';
  // 默认展开前 2 层
  const depth = (path.match(/\./g) || []).length;
  if (depth >= 2) {
    children.style.display = 'none';
    header.querySelector('.jt-tree-toggle').textContent = '▶';
  } else {
    header.querySelector('.jt-tree-toggle').textContent = '▼';
  }

  for (const k of childKeys) {
    const childPath = isArray ? `${path}[${k}]` : `${path}.${k}`;
    children.appendChild(renderNode(value[k], k, childPath));
  }

  wrap.appendChild(header);
  wrap.appendChild(children);
  return wrap;
}

function toggleNode(btn) {
  const children = btn.closest('.jt-tree-header').nextSibling;
  const isHidden = children.style.display === 'none';
  children.style.display = isHidden ? '' : 'none';
  btn.textContent = isHidden ? '▼' : '▶';
}

function expandAll()   { document.querySelectorAll('.jt-tree-children').forEach(n => n.style.display = ''); }
function collapseAll() { document.querySelectorAll('.jt-tree-children').forEach(n => n.style.display = 'none'); }

function copyPath(path) {
  navigator.clipboard.writeText(path).then(() => showToast(`已复制：${path}`, 'success'));
}

function renderValue(v) {
  if (v === null)           return '<span class="jt-val-null">null</span>';
  if (typeof v === 'boolean') return `<span class="jt-val-bool">${v}</span>`;
  if (typeof v === 'number')  return `<span class="jt-val-num">${v}</span>`;
  return `<span class="jt-val-str">"${escapeHtml(String(v))}"</span>`;
}
```

## 3. Table Viewer JS（`/static/js/json-tool-table.js`）

```javascript
function processJson() {
  const parsed = parseInput();
  if (parsed === null) return;

  if (!Array.isArray(parsed)) {
    showToast('输入必须为 JSON 数组', 'error'); return;
  }

  // 收集所有 Key
  const keys = [...new Set(parsed.flatMap(row =>
    typeof row === 'object' && row ? Object.keys(row) : []
  ))];

  renderTable(parsed, keys);
}

let tableData = [];
let sortState = { key: null, dir: 'asc' };

function renderTable(data, keys) {
  tableData = data;
  const container = document.getElementById('tableOutput');
  const searchVal = document.getElementById('tableSearch')?.value.toLowerCase() || '';
  const filtered  = searchVal
    ? data.filter(row => JSON.stringify(row).toLowerCase().includes(searchVal))
    : data;

  const thead = keys.map(k => `
    <th onclick="sortTable('${escapeHtml(k)}')" class="jt-table-th">
      ${escapeHtml(k)}
      ${sortState.key === k ? (sortState.dir === 'asc' ? ' ↑' : ' ↓') : ''}
    </th>
  `).join('');

  const tbody = filtered.map(row => `
    <tr>${keys.map(k => `<td class="jt-table-td">${escapeHtml(String(row[k] ?? ''))}</td>`).join('')}</tr>
  `).join('');

  container.innerHTML = `
    <div class="jt-table-toolbar">
      <input type="text" id="tableSearch" placeholder="搜索..." oninput="processJson()"
             class="jt-table-search" value="${searchVal}">
      <span class="jt-table-count">共 ${filtered.length} 行</span>
    </div>
    <div class="jt-table-wrap">
      <table class="jt-table"><thead><tr>${thead}</tr></thead><tbody>${tbody}</tbody></table>
    </div>
  `;
}

function sortTable(key) {
  const dir = sortState.key === key && sortState.dir === 'asc' ? 'desc' : 'asc';
  sortState = { key, dir };
  tableData.sort((a, b) => {
    const va = String(a[key] ?? '');
    const vb = String(b[key] ?? '');
    return dir === 'asc' ? va.localeCompare(vb) : vb.localeCompare(va);
  });
  const keys = [...new Set(tableData.flatMap(r => Object.keys(r)))];
  renderTable(tableData, keys);
}
```

```javascript
function processJson() {
  const raw    = getInput().trim();
  const parsed = parseInput();
  if (parsed === null) return;

  const stats  = analyzeSize(parsed);
  const total  = new Blob([raw]).size;

  document.getElementById('sizeOutput').innerHTML = `
    <div class="jt-size-grid">
      <div class="jt-size-stat"><span class="jt-size-label">总大小（原始）</span><span class="jt-size-value">${formatBytes(total)}</span></div>
      <div class="jt-size-stat"><span class="jt-size-label">总大小（压缩）</span><span class="jt-size-value">${formatBytes(new Blob([JSON.stringify(parsed)]).size)}</span></div>
      <div class="jt-size-stat"><span class="jt-size-label">最大深度</span><span class="jt-size-value">${stats.maxDepth}</span></div>
      <div class="jt-size-stat"><span class="jt-size-label">总节点数</span><span class="jt-size-value">${stats.nodes}</span></div>
      <div class="jt-size-stat"><span class="jt-size-label">对象数</span><span class="jt-size-value">${stats.objects}</span></div>
      <div class="jt-size-stat"><span class="jt-size-label">数组数</span><span class="jt-size-value">${stats.arrays}</span></div>
      <div class="jt-size-stat"><span class="jt-size-label">字符串数</span><span class="jt-size-value">${stats.strings}</span></div>
      <div class="jt-size-stat"><span class="jt-size-label">数字数</span><span class="jt-size-value">${stats.numbers}</span></div>
      <div class="jt-size-stat"><span class="jt-size-label">Key 总数</span><span class="jt-size-value">${stats.keys}</span></div>
      <div class="jt-size-stat"><span class="jt-size-label">null 值数</span><span class="jt-size-value">${stats.nulls}</span></div>
    </div>
  `;
}

function analyzeSize(node, depth = 0) {
  const acc = { nodes:0, objects:0, arrays:0, strings:0, numbers:0, nulls:0, booleans:0, keys:0, maxDepth:depth };
  function walk(v, d) {
    acc.nodes++;
    acc.maxDepth = Math.max(acc.maxDepth, d);
    if (v === null)             { acc.nulls++; return; }
    if (typeof v === 'boolean') { acc.booleans++; return; }
    if (typeof v === 'number')  { acc.numbers++; return; }
    if (typeof v === 'string')  { acc.strings++; return; }
    if (Array.isArray(v))       { acc.arrays++;  v.forEach(i => walk(i, d+1)); return; }
    if (typeof v === 'object')  {
      acc.objects++;
      const k = Object.keys(v);
      acc.keys += k.length;
      k.forEach(key => walk(v[key], d+1));
    }
  }
  walk(node, depth);
  return acc;
}
```

## 5. Flatten JS（`/static/js/json-tool-flatten.js`）

```javascript
function processJson() {
  const parsed    = parseInput();
  if (parsed === null) return;
  const delimiter = document.getElementById('flatDelimiter')?.value || '.';
  const mode      = document.querySelector('input[name="flatMode"]:checked')?.value || 'flatten';

  if (mode === 'flatten') {
    setOutput(JSON.stringify(flattenJson(parsed, delimiter), null, 2));
  } else {
    setOutput(JSON.stringify(unflattenJson(parsed, delimiter), null, 2));
  }
}

function flattenJson(obj, delimiter = '.', prefix = '') {
  const result = {};
  for (const [k, v] of Object.entries(obj)) {
    const newKey = prefix ? `${prefix}${delimiter}${k}` : k;
    if (typeof v === 'object' && v !== null && !Array.isArray(v)) {
      Object.assign(result, flattenJson(v, delimiter, newKey));
    } else {
      result[newKey] = v;
    }
  }
  return result;
}

function unflattenJson(flat, delimiter = '.') {
  const result = {};
  for (const [k, v] of Object.entries(flat)) {
    const keys = k.split(delimiter);
    let cur = result;
    for (let i = 0; i < keys.length - 1; i++) {
      if (!(keys[i] in cur)) cur[keys[i]] = {};
      cur = cur[keys[i]];
    }
    cur[keys[keys.length-1]] = v;
  }
  return result;
}
```
