```javascript
function encodeJson() {
  const raw = getInput().trim();
  if (!raw) return;
  try {
    JSON.parse(raw); // 验证是合法 JSON
    setOutput(btoa(unescape(encodeURIComponent(raw))), 'plaintext');
  } catch(e) { showErrorPanel(e, raw); }
}

function decodeJson() {
  const raw = getInput().trim();
  if (!raw) return;
  try {
    const decoded = decodeURIComponent(escape(atob(raw)));
    const parsed  = JSON.parse(decoded);
    setOutput(JSON.stringify(parsed, null, 2));
  } catch(e) {
    showToast('Base64 解码失败，请确认输入为合法 Base64 编码的 JSON', 'error');
  }
}
```

## 2. JWT Decoder JS（`/static/js/json-tool-jwt.js`）

```javascript
function processJson() {
  const raw    = getInput().trim().replace(/^Bearer\s+/i, '');
  if (!raw) return;

  const parts  = raw.split('.');
  if (parts.length < 2) { showToast('无效的 JWT Token（需含至少 2 段）', 'error'); return; }

  try {
    const header  = JSON.parse(atob(parts[0].replace(/-/g,'+').replace(/_/g,'/')));
    const payload = JSON.parse(atob(parts[1].replace(/-/g,'+').replace(/_/g,'/')));

    // 过期检查
    const now = Math.floor(Date.now() / 1000);
    const exp = payload.exp;
    const iat = payload.iat;
    const isExpired = exp && exp < now;

    const result = {
      header,
      payload,
      signature: parts[2] || '',
      meta: {
        algorithm:  header.alg,
        type:       header.typ,
        issuedAt:   iat  ? new Date(iat  * 1000).toISOString() : null,
        expiresAt:  exp  ? new Date(exp  * 1000).toISOString() : null,
        isExpired,
        timeLeft:   exp && !isExpired ? formatDuration(exp - now) : null,
      }
    };

    setOutput(JSON.stringify(result, null, 2));
    renderJwtPanel(result);
  } catch(e) {
    showToast('JWT 解析失败：' + e.message, 'error');
  }
}

function renderJwtPanel(jwt) {
  const panel = document.getElementById('jwtPanel');
  if (!panel) return;
  const { meta } = jwt;
  panel.innerHTML = `
    <div class="jt-jwt-grid">
      <div class="jt-jwt-section">
        <h3>Header</h3>
        <pre>${escapeHtml(JSON.stringify(jwt.header, null, 2))}</pre>
      </div>
      <div class="jt-jwt-section">
        <h3>Payload</h3>
        <pre>${escapeHtml(JSON.stringify(jwt.payload, null, 2))}</pre>
      </div>
    </div>
    <div class="jt-jwt-meta">
      <span>算法：<strong>${meta.algorithm}</strong></span>
      ${meta.issuedAt  ? `<span>签发：${meta.issuedAt}</span>` : ''}
      ${meta.expiresAt ? `<span class="${meta.isExpired ? 'jt-jwt-expired':''}">
        ${meta.isExpired ? '⚠️ 已过期' : `有效期至：${meta.expiresAt}（剩余 ${meta.timeLeft}）`}
      </span>` : ''}
    </div>
  `;
  panel.style.display = 'block';
}

function formatDuration(seconds) {
  if (seconds < 60)          return `${seconds} 秒`;
  if (seconds < 3600)        return `${Math.floor(seconds/60)} 分钟`;
  if (seconds < 86400)       return `${Math.floor(seconds/3600)} 小时`;
  return `${Math.floor(seconds/86400)} 天`;
}
```

## 3. JSONC 工具 JS（`/static/js/json-tool-jsonc.js`）

```javascript
// 移除 JSONC/JSON5 注释，转为标准 JSON
function processJson() {
  const raw = getInput().trim();
  if (!raw) return;

  // 移除单行注释（// ...）和多行注释（/* ... */），保留字符串内部内容
  const stripped = stripJsonComments(raw);

  try {
    const parsed = JSON.parse(stripped);
    setOutput(JSON.stringify(parsed, null, 2));
    showToast('✅ 注释移除成功', 'success');
  } catch(e) {
    showErrorPanel(e, stripped);
  }
}

function stripJsonComments(str) {
  let result   = '';
  let inString = false;
  let escape   = false;
  let i        = 0;

  while (i < str.length) {
    const ch  = str[i];
    const ch2 = str[i + 1];

    if (escape)            { result += ch; escape = false; i++; continue; }
    if (ch === '\\' && inString) { result += ch; escape = true; i++; continue; }
    if (ch === '"')        { inString = !inString; result += ch; i++; continue; }
    if (inString)          { result += ch; i++; continue; }

    // 单行注释
    if (ch === '/' && ch2 === '/') {
      while (i < str.length && str[i] !== '\n') i++;
      continue;
    }
    // 多行注释
    if (ch === '/' && ch2 === '*') {
      i += 2;
      while (i < str.length && !(str[i] === '*' && str[i+1] === '/')) i++;
      i += 2;
      continue;
    }
    // 尾随逗号（JSON5）
    if (ch === ',' && /\s*[}\]]/.test(str.slice(i+1, i+10))) {
      i++; continue;
    }

    result += ch;
    i++;
  }
  return result;
}
```

## 4. Token Counter 后端（`/json/api/token-count`）

```go
// internal/handler/json_token.go
// 依赖：github.com/pkoukk/tiktoken-go

package handler

import (
    "github.com/pkoukk/tiktoken-go"
    "strings"
    "unicode"
)

var modelEncodings = map[string]string{
    "gpt-4":            "cl100k_base",
    "gpt-4o":           "o200k_base",
    "gpt-3.5-turbo":    "cl100k_base",
    "claude-3":         "cl100k_base", // 近似
    "claude-3-5":       "cl100k_base",
    "gemini-1.5-pro":   "cl100k_base", // 近似
    "text-embedding-3": "cl100k_base",
}

func JsonTokenCount(c *gin.Context) {
    var req struct {
        Text  string `json:"text"`
        Model string `json:"model"`
    }
    if err := c.ShouldBindJSON(&req); err != nil {
        c.JSON(400, gin.H{"error": "invalid request"})
        return
    }

    encName, ok := modelEncodings[req.Model]
    if !ok { encName = "cl100k_base" }

    enc, err := tiktoken.GetEncoding(encName)
    if err != nil {
        c.JSON(500, gin.H{"error": "tokenizer load failed"})
        return
    }

    tokens := enc.Encode(req.Text, nil, nil)
    c.JSON(200, gin.H{
        "count": len(tokens),
        "model": req.Model,
        "chars": len([]rune(req.Text)),
        "bytes": len(req.Text),
        "words": wordCount(req.Text),
    })
}

func wordCount(s string) int {
    count  := 0
    inWord := false
    for _, r := range s {
        if unicode.IsSpace(r) { inWord = false } else if !inWord { inWord = true; count++ }
    }
    return count
}
```

### Token Counter 前端 JS（`/static/js/json-tool-token-count.js`）

```javascript
async function processJson() {
  const text  = getInput();
  if (!text.trim()) return;
  const model = document.getElementById('modelSelect')?.value || 'gpt-4';

  const btn = document.getElementById('processBtn');
  btn.disabled    = true;
  btn.textContent = '计算中...';

  try {
    const resp = await fetch('/json/api/token-count', {
      method:  'POST',
      headers: { 'Content-Type': 'application/json' },
      body:    JSON.stringify({ text, model }),
    });
    const data = await resp.json();

    document.getElementById('tokenCount').textContent = data.count.toLocaleString();
    document.getElementById('charCount').textContent  = data.chars.toLocaleString();
    document.getElementById('wordCount').textContent  = data.words.toLocaleString();
    document.getElementById('byteCount').textContent  = formatBytes(data.bytes);

    // 警告阈值
    const limits = { 'gpt-4':128000, 'gpt-3.5-turbo':16385, 'claude-3':200000 };
    const limit  = limits[model];
    if (limit && data.count > limit * 0.9) {
      showToast(`⚠️ 接近 ${model} 上下文限制（${limit.toLocaleString()} Token）`, 'error');
    }
  } catch(e) {
    showToast('Token 计数失败：' + e.message, 'error');
  } finally {
    btn.disabled    = false;
    btn.textContent = '统计';
  }
}
```
