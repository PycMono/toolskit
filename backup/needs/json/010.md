```javascript
// 类型推断工具
function inferType(value, lang) {
  if (value === null)           return nullType(lang);
  if (typeof value === 'boolean') return boolType(lang);
  if (typeof value === 'number')
    return Number.isInteger(value) ? intType(lang) : floatType(lang);
  if (typeof value === 'string')  return stringType(lang);
  if (Array.isArray(value)) {
    const itemType = value.length > 0 ? inferType(value[0], lang) : anyType(lang);
    return arrayType(lang, itemType);
  }
  return objectType(lang);
}

// 各语言类型名映射
const typeMap = {
  null:   { ts:'null',    py:'None',    java:'Object', cs:'object',  go:'interface{}' },
  bool:   { ts:'boolean', py:'bool',    java:'boolean',cs:'bool',    go:'bool' },
  int:    { ts:'number',  py:'int',     java:'int',    cs:'int',     go:'int' },
  float:  { ts:'number',  py:'float',   java:'double', cs:'double',  go:'float64' },
  string: { ts:'string',  py:'str',     java:'String', cs:'string',  go:'string' },
  any:    { ts:'any',     py:'Any',     java:'Object', cs:'object',  go:'interface{}' },
};

function nullType(l)        { return typeMap.null[l]; }
function boolType(l)        { return typeMap.bool[l]; }
function intType(l)         { return typeMap.int[l]; }
function floatType(l)       { return typeMap.float[l]; }
function stringType(l)      { return typeMap.string[l]; }
function anyType(l)         { return typeMap.any[l]; }
function arrayType(l, item) {
  if (l === 'ts')   return `${item}[]`;
  if (l === 'py')   return `List[${item}]`;
  if (l === 'java') return `List<${item}>`;
  if (l === 'cs')   return `List<${item}>`;
  if (l === 'go')   return `[]${item}`;
  return `${item}[]`;
}
function objectType(l) { return typeMap.any[l]; }

function toPascalCase(str) {
  return str.replace(/[-_\s]+(.)/g, (_, c) => c.toUpperCase())
            .replace(/^(.)/, c => c.toUpperCase());
}

function toCamelCase(str) {
  const p = toPascalCase(str);
  return p.charAt(0).toLowerCase() + p.slice(1);
}
```

## 2. 各语言生成 JS

### TypeScript（`/static/js/json-tool-to-typescript.js`）

```javascript
function processJson() {
  const parsed    = parseInput();
  if (parsed === null) return;
  const rootName  = toPascalCase(document.getElementById('className')?.value || 'Root');
  const interfaces = [];

  function genInterface(obj, name) {
    if (typeof obj !== 'object' || obj === null || Array.isArray(obj)) return;
    const fields = Object.entries(obj).map(([k, v]) => {
      if (typeof v === 'object' && v !== null && !Array.isArray(v)) {
        genInterface(v, toPascalCase(k));
        return `  ${k}: ${toPascalCase(k)};`;
      }
      if (Array.isArray(v) && v.length > 0 && typeof v[0] === 'object') {
        genInterface(v[0], toPascalCase(k) + 'Item');
        return `  ${k}: ${toPascalCase(k)}Item[];`;
      }
      return `  ${k}: ${inferType(v, 'ts')};`;
    }).join('\n');
    interfaces.push(`export interface ${name} {\n${fields}\n}`);
  }

  genInterface(Array.isArray(parsed) ? parsed[0] : parsed, rootName);
  setOutput(interfaces.reverse().join('\n\n'), 'typescript');
}
```

### Python（`/static/js/json-tool-to-python.js`）

```javascript
function processJson() {
  const parsed   = parseInput();
  if (parsed === null) return;
  const rootName = toPascalCase(document.getElementById('className')?.value || 'Root');
  const classes  = [];

  function genClass(obj, name) {
    if (typeof obj !== 'object' || obj === null || Array.isArray(obj)) return;
    const fields = Object.entries(obj).map(([k, v]) => {
      if (typeof v === 'object' && v !== null && !Array.isArray(v)) {
        genClass(v, toPascalCase(k));
        return `    ${k}: ${toPascalCase(k)}`;
      }
      return `    ${k}: ${inferType(v, 'py')}`;
    }).join('\n');
    classes.push(`@dataclass\nclass ${name}:\n${fields}`);
  }

  genClass(Array.isArray(parsed) ? parsed[0] : parsed, rootName);
  const imports = 'from dataclasses import dataclass\nfrom typing import List, Optional, Any\n\n';
  setOutput(imports + classes.reverse().join('\n\n'), 'python');
}
```

### Go（`/static/js/json-tool-to-go.js`）

```javascript
function processJson() {
  const parsed   = parseInput();
  if (parsed === null) return;
  const rootName = toPascalCase(document.getElementById('className')?.value || 'Root');
  const structs  = [];

  function genStruct(obj, name) {
    if (typeof obj !== 'object' || obj === null || Array.isArray(obj)) return;
    const fields = Object.entries(obj).map(([k, v]) => {
      const fieldName = toPascalCase(k);
      const jsonTag   = `\`json:"${k}"\``;
      if (typeof v === 'object' && v !== null && !Array.isArray(v)) {
        genStruct(v, toPascalCase(k));
        return `\t${fieldName} ${toPascalCase(k)} ${jsonTag}`;
      }
      if (Array.isArray(v) && v.length > 0 && typeof v[0] === 'object') {
        genStruct(v[0], toPascalCase(k));
        return `\t${fieldName} []${toPascalCase(k)} ${jsonTag}`;
      }
      return `\t${fieldName} ${inferType(v, 'go')} ${jsonTag}`;
    }).join('\n');
    structs.push(`type ${name} struct {\n${fields}\n}`);
  }

  genStruct(Array.isArray(parsed) ? parsed[0] : parsed, rootName);
  setOutput('package main\n\n' + structs.reverse().join('\n\n'), 'go');
}
```
