## 1. Schema Validator JS（`/static/js/json-tool-schema-validate.js`）

```javascript
// 双编辑器：上方 Schema，下方待验证 JSON
// 依赖 ajv（CDN 加载）

let schemaEditor = null;

function initSchemaEditor() {
  require(['vs/editor/editor.main'], function() {
    schemaEditor = monaco.editor.create(document.getElementById('schemaEditor'), {
      value:          JSON.stringify({ "$schema": "http://json-schema.org/draft-07/schema#", "type": "object" }, null, 2),
      language:       'json',
      fontSize:       13,
      minimap:        { enabled: false },
      automaticLayout:true,
    });
  });
}

function processJson() {
  const schemaRaw = schemaEditor?.getValue().trim();
  const dataRaw   = getInput().trim();

  let schema, data;
  try { schema = JSON.parse(schemaRaw); } catch(e) { showToast('Schema 格式错误', 'error'); return; }
  try { data   = JSON.parse(dataRaw);   } catch(e) { showErrorPanel(e, dataRaw); return; }

  const Ajv    = window.ajv2020 || window.Ajv;
  const ajv    = new Ajv({ allErrors: true });
  const valid  = ajv.validate(schema, data);

  if (valid) {
    setOutput('✅ 验证通过', 'plaintext');
    showToast('✅ JSON 验证通过', 'success');
  } else {
    const errors = ajv.errors.map(e =>
      `❌ ${e.instancePath || '(root)'} ${e.message}`
    ).join('\n');
    setOutput(errors, 'plaintext');
    showToast(`验证失败，${ajv.errors.length} 个错误`, 'error');
  }
}

document.addEventListener('DOMContentLoaded', initSchemaEditor);
```

## 2. Schema Generator JS（`/static/js/json-tool-schema-generate.js`）

```javascript
function processJson() {
  const parsed = parseInput();
  if (parsed === null) return;
  const schema = generateSchema(parsed);
  setOutput(JSON.stringify(schema, null, 2));
}

function generateSchema(value, title) {
  if (value === null) return { type: ['null', 'string'] };

  if (Array.isArray(value)) {
    const itemSchema = value.length > 0 ? generateSchema(value[0]) : {};
    return { type: 'array', items: itemSchema };
  }

  if (typeof value === 'object') {
    const properties = {};
    const required   = [];
    for (const [k, v] of Object.entries(value)) {
      properties[k] = generateSchema(v);
      if (v !== null && v !== undefined) required.push(k);
    }
    const schema = {
      type: 'object',
      properties,
    };
    if (required.length > 0) schema.required = required;
    if (title) schema.title = title;
    return schema;
  }

  if (typeof value === 'boolean') return { type: 'boolean' };
  if (typeof value === 'number')
    return Number.isInteger(value) ? { type: 'integer' } : { type: 'number' };
  if (typeof value === 'string') {
    const schema = { type: 'string' };
    // 尝试检测格式
    if (/^\d{4}-\d{2}-\d{2}/.test(value))   schema.format = 'date-time';
    if (/^[^@\s]+@[^@\s]+\.[^@\s]+$/.test(value)) schema.format = 'email';
    if (/^https?:\/\//.test(value))          schema.format = 'uri';
    return schema;
  }
  return {};
}
```

## 3. 验收标准

- [ ] Schema Validate：上传合法 Schema + 合法 JSON → 显示「验证通过」
- [ ] Schema Validate：不符合 Schema 的 JSON → 显示具体错误路径
- [ ] Schema Generate：`{"name":"Alice","age":30}` 生成 `{"type":"object","properties":{"name":{"type":"string"},"age":{"type":"integer"}},"required":["name","age"]}`
- [ ] Schema Generate：含 URL/Email 字段时自动添加 `format` 注解
