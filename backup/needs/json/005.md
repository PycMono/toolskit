## 2. JS 实现

### repair 工具（`/static/js/json-tool-repair.js`）

```javascript
// JSON 自动修复：处理常见错误

function processJson() {
  const raw = getInput().trim();
  if (!raw) return;

  const repaired = repairJson(raw);
  setOutput(repaired);

  // 显示修复统计
  const changes = countRepairChanges(raw, repaired);
  const el = document.getElementById('outputStats');
  if (el) el.textContent = changes > 0 ? `修复了 ${changes} 处问题` : '✅ JSON 无需修复';
}

function repairJson(raw) {
  let s = raw.trim();

  // 1. 移除 BOM
  s = s.replace(/^\uFEFF/, '');

  // 2. 移除 JS 注释（// 和 /* */）
  s = s.replace(/\/\/[^\n]*/g, '').replace(/\/\*[\s\S]*?\*\//g, '');

  // 3. 替换单引号为双引号（简单场景）
  s = s.replace(/([{,]\s*)'([^']+)'\s*:/g, '$1"$2":');
  s = s.replace(/:\s*'([^']*)'/g, ': "$1"');

  // 4. 移除尾随逗号
  s = s.replace(/,(\s*[}\]])/g, '$1');

  // 5. 补全缺失的引号（Key 未加引号）
  s = s.replace(/([{,]\s*)([a-zA-Z_$][a-zA-Z0-9_$]*)\s*:/g, '$1"$2":');

  // 6. undefined → null
  s = s.replace(/:\s*undefined/g, ': null');

  // 7. NaN / Infinity → null
  s = s.replace(/:\s*NaN/g, ': null');
  s = s.replace(/:\s*Infinity/g, ': null');

  // 8. 尝试 parse，若仍失败则尝试补全括号
  try {
    JSON.parse(s);
    return JSON.stringify(JSON.parse(s), null, 2);
  } catch(e) {
    // 尝试补全缺失括号
    s = balanceBrackets(s);
    try {
      return JSON.stringify(JSON.parse(s), null, 2);
    } catch(e2) {
      showToast('修复失败，JSON 结构损坏严重', 'error');
      return s;
    }
  }
}

function balanceBrackets(s) {
  const opens   = ['{','['];
  const closes  = ['}',']'];
  const pairs   = {'{':'}','[':']'};
  const stack   = [];
  for (const ch of s) {
    if (opens.includes(ch))  stack.push(ch);
    if (closes.includes(ch)) {
      if (stack.length && pairs[stack[stack.length-1]] === ch) stack.pop();
    }
  }
  while (stack.length) s += pairs[stack.pop()];
  return s;
}

function countRepairChanges(original, repaired) {
  let count = 0;
  if (original !== repaired) count++;
  return count; // 简化实现
}
```

### escape / unescape / stringify 工具

```javascript
// /static/js/json-tool-escape.js
function processJson() {
  const raw = getInput().trim();
  if (!raw) return;
  try {
    JSON.parse(raw);  // 验证输入是合法 JSON
    const escaped = raw
      .replace(/\\/g, '\\\\')
      .replace(/"/g, '\\"')
      .replace(/\n/g, '\\n')
      .replace(/\r/g, '\\r')
      .replace(/\t/g, '\\t');
    setOutput(escaped, 'plaintext');
  } catch(e) {
    showErrorPanel(e, raw);
  }
}

// /static/js/json-tool-unescape.js
function processJson() {
  const raw = getInput().trim();
  if (!raw) return;
  try {
    const unescaped = raw
      .replace(/\\n/g, '\n')
      .replace(/\\r/g, '\r')
      .replace(/\\t/g, '\t')
      .replace(/\\"/g, '"')
      .replace(/\\\\/g, '\\');
    JSON.parse(unescaped); // 验证结果
    setOutput(JSON.stringify(JSON.parse(unescaped), null, 2));
  } catch(e) {
    showToast('反转义结果不是合法 JSON', 'error');
  }
}

// /static/js/json-tool-stringify.js
function processJson() {
  const parsed = parseInput();
  if (parsed === null) return;
  const stringified = JSON.stringify(JSON.stringify(parsed));
  setOutput(stringified, 'plaintext');
}
```
