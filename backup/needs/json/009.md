```javascript
function handleExcelUpload(input) {
  const file = input.files[0];
  if (!file) return;

  const reader = new FileReader();
  reader.onload = e => {
    const data = new Uint8Array(e.target.result);
    const wb   = XLSX.read(data, { type: 'array' });
    const sheetName = wb.SheetNames[0];
    const ws        = wb.Sheets[sheetName];
    const json      = XLSX.utils.sheet_to_json(ws, { defval: null });
    setOutput(JSON.stringify(json, null, 2));
  };
  reader.readAsArrayBuffer(file);
  input.value = '';
}
```

### JSON вҶ’ SQLпјҲ`/static/js/json-tool-to-sql.js`пјү

```javascript
function processJson() {
  const parsed    = parseInput();
  if (parsed === null) return;
  if (!Array.isArray(parsed)) { showToast('иҫ“е…Ҙеҝ…йЎ»дёә JSON ж•°з»„', 'error'); return; }

  const table     = document.getElementById('sqlTableName')?.value || 'table_name';
  const dialect   = document.getElementById('sqlDialect')?.value   || 'mysql';

  const keys     = [...new Set(parsed.flatMap(r => Object.keys(r)))];
  const quote    = dialect === 'mssql' ? '[' : '`';
  const quoteEnd = dialect === 'mssql' ? ']' : '`';

  const colDefs  = keys.map(k => `  ${quote}${k}${quoteEnd} TEXT`).join(',\n');
  const createSql = `CREATE TABLE ${quote}${table}${quoteEnd} (\n${colDefs}\n);\n\n`;

  const inserts = parsed.map(row => {
    const vals = keys.map(k => {
      const v = row[k];
      if (v === null || v === undefined) return 'NULL';
      if (typeof v === 'number')         return String(v);
      return `'${String(v).replace(/'/g, "''")}'`;
    });
    return `INSERT INTO ${quote}${table}${quoteEnd} (${keys.map(k=>`${quote}${k}${quoteEnd}`).join(', ')}) VALUES (${vals.join(', ')});`;
  }).join('\n');

  setOutput(createSql + inserts, 'sql');
}
```

```javascript
// и§Јжһҗ INSERT INTO иҜӯеҸҘ вҶ’ JSON ж•°з»„
function processJson() {
  const raw = getInput().trim();
  if (!raw) return;

  const insertRegex = /INSERT\s+INTO\s+[`\[\]"']?\w+[`\[\]"']?\s*\(([^)]+)\)\s*VALUES\s*\(([^)]+)\)/gi;
  const results = [];
  let match;

  while ((match = insertRegex.exec(raw)) !== null) {
    const keys = match[1].split(',').map(k => k.trim().replace(/[`\[\]"']/g,''));
    const vals = parseValues(match[2]);
    const row  = {};
    keys.forEach((k, i) => {
      const v = vals[i];
      row[k] = v === 'NULL' ? null : isNaN(v) ? v : Number(v);
    });
    results.push(row);
  }

  if (results.length === 0) { showToast('жңӘжүҫеҲ° INSERT иҜӯеҸҘ', 'error'); return; }
  setOutput(JSON.stringify(results, null, 2));
}

function parseValues(str) {
  const vals = [];
  let cur = '', inQuote = false, quote = '';
  for (const ch of str) {
    if (!inQuote && (ch === "'" || ch === '"')) { inQuote = true; quote = ch; }
    else if (inQuote && ch === quote)            { inQuote = false; vals.push(cur); cur = ''; }
    else if (!inQuote && ch === ',')             { vals.push(cur.trim()); cur = ''; }
    else if (inQuote || ch !== ' ')              { cur += ch; }
  }
  if (cur.trim()) vals.push(cur.trim());
  return vals;
}
```

### JSON вҶ’ MarkdownпјҲ`/static/js/json-tool-to-markdown.js`пјү

```javascript
function processJson() {
  const parsed = parseInput();
  if (parsed === null) return;
  if (!Array.isArray(parsed)) { showToast('иҫ“е…Ҙеҝ…йЎ»дёә JSON ж•°з»„', 'error'); return; }

  const keys = [...new Set(parsed.flatMap(r => typeof r === 'object' ? Object.keys(r) : []))];
  const header   = `| ${keys.join(' | ')} |`;
  const divider  = `| ${keys.map(() => '---').join(' | ')} |`;
  const rows     = parsed.map(row =>
    `| ${keys.map(k => String(row[k] ?? '').replace(/\|/g, '\\|')).join(' | ')} |`
  );

  const md = [header, divider, ...rows].join('\n');
  setOutput(md, 'markdown');
}
```
