## 3. 首页 JS（`/static/js/json-home.js`）

```javascript
// /static/js/json-home.js

// 工具搜索过滤
function filterTools(query) {
  const q = query.trim().toLowerCase();
  document.querySelectorAll('.jt-tool-card').forEach(card => {
    const text = card.textContent.toLowerCase();
    card.style.display = (!q || text.includes(q)) ? '' : 'none';
  });
  // 隐藏空分组
  document.querySelectorAll('.jt-tool-group').forEach(group => {
    const visible = [...group.querySelectorAll('.jt-tool-card')]
      .some(c => c.style.display !== 'none');
    group.style.display = visible ? '' : 'none';
  });
}

// 最近使用工具（localStorage）
const RECENT_KEY = 'jt_recent_tools';
const MAX_RECENT = 6;

function loadRecentTools() {
  try {
    const recent = JSON.parse(localStorage.getItem(RECENT_KEY) || '[]');
    if (recent.length === 0) return;
    const container = document.getElementById('recentList');
    const section   = document.getElementById('recentTools');
    container.innerHTML = recent.map(t => `
      <a class="jt-recent-chip" href="/json/${t.key}">
        <span>${t.icon}</span> ${t.name}
      </a>
    `).join('');
    section.style.display = 'block';
  } catch(e) {}
}

function recordToolVisit(key, icon, name) {
  try {
    let recent = JSON.parse(localStorage.getItem(RECENT_KEY) || '[]');
    recent     = recent.filter(t => t.key !== key);
    recent.unshift({ key, icon, name });
    recent     = recent.slice(0, MAX_RECENT);
    localStorage.setItem(RECENT_KEY, JSON.stringify(recent));
  } catch(e) {}
}

document.addEventListener('DOMContentLoaded', loadRecentTools);
```
