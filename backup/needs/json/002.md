// 通用 Handler 工厂
func JsonToolPage(toolKey string) gin.HandlerFunc {
    return func(c *gin.Context) {
        lang  := c.GetString("lang")
        meta  := getJsonToolMeta(toolKey)
        title := meta.TitleZH
        desc  := meta.DescZH
        if lang == "en" {
            title = meta.TitleEN
            desc  = meta.DescEN
        }
        siteTitle := map[string]string{
            "zh": title + " — 免费在线 JSON 工具 | DevToolBox",
            "en": title + " — Free Online JSON Tool | DevToolBox",
        }
        c.HTML(200, "json/tool.html", gin.H{
            "Lang":    lang,
            "Title":   siteTitle[lang],
            "Desc":    desc,
            "ToolKey": toolKey,
            "Meta":    meta,
            "Path":    "/json/" + toolKey,
            "FAQs":    getJsonToolFAQs(toolKey, lang),
            "AllTools": jsonToolMetas,
        })
    }
}

// Token Count 后端 API（唯一后端接口）
func JsonTokenCount(c *gin.Context) {
    var req struct {
        Text  string `json:"text"`
        Model string `json:"model"` // gpt-4 / claude-3 / gemini-1.5-pro 等
    }
    if err := c.ShouldBindJSON(&req); err != nil {
        c.JSON(400, gin.H{"error": "invalid request"})
        return
    }
    // 调用 tiktoken-go 或自行实现 tokenizer
    count, model, err := tokenizer.Count(req.Text, req.Model)
    if err != nil {
        c.JSON(500, gin.H{"error": err.Error()})
        return
    }
    c.JSON(200, gin.H{
        "count": count,
        "model": model,
        "chars": len([]rune(req.Text)),
        "bytes": len(req.Text),
        "words": wordCount(req.Text),
    })
}
