## 1. JS 实现

### Kotlin（`/static/js/json-tool-to-kotlin.js`）

```javascript
function processJson() {
  const parsed   = parseInput();
  if (parsed === null) return;
  const rootName = toPascalCase(document.getElementById('className')?.value || 'Root');
  const classes  = [];

  function genDataClass(obj, name) {
    if (typeof obj !== 'object' || obj === null || Array.isArray(obj)) return;
    const fields = Object.entries(obj).map(([k, v]) => {
      const prop = toCamelCase(k);
      const ann  = k !== prop ? `@SerialName("${k}") ` : '';
      if (typeof v === 'object' && v !== null && !Array.isArray(v)) {
        genDataClass(v, toPascalCase(k));
        return `  ${ann}val ${prop}: ${toPascalCase(k)}`;
      }
      if (Array.isArray(v) && v.length > 0 && typeof v[0] === 'object') {
        genDataClass(v[0], toPascalCase(k));
        return `  ${ann}val ${prop}: List<${toPascalCase(k)}>`;
      }
      const type = inferType(v, 'ts')  // 借用 ts 映射后手动调整
        .replace('boolean','Boolean').replace('number',Number.isInteger(v)?'Int':'Double')
        .replace('string','String');
      return `  ${ann}val ${prop}: ${type}`;
    }).join(',\n');
    classes.push(`@Serializable\ndata class ${name}(\n${fields}\n)`);
  }

  genDataClass(Array.isArray(parsed) ? parsed[0] : parsed, rootName);
  const imports = 'import kotlinx.serialization.*\nimport kotlinx.serialization.json.*\n\n';
  setOutput(imports + classes.reverse().join('\n\n'), 'kotlin');
}
```

### Swift（`/static/js/json-tool-to-swift.js`）

```javascript
function processJson() {
  const parsed   = parseInput();
  if (parsed === null) return;
  const rootName = toPascalCase(document.getElementById('className')?.value || 'Root');
  const structs  = [];

  function genStruct(obj, name) {
    if (typeof obj !== 'object' || obj === null || Array.isArray(obj)) return;
    const fields = Object.entries(obj).map(([k, v]) => {
      const prop = toCamelCase(k);
      if (typeof v === 'object' && v !== null && !Array.isArray(v)) {
        genStruct(v, toPascalCase(k));
        return `    let ${prop}: ${toPascalCase(k)}`;
      }
      const swiftType = { boolean:'Bool', number:'Double', string:'String', null:'String?' };
      const t = typeof v === 'object' ? 'String?' : (swiftType[typeof v] || 'String');
      return `    let ${prop}: ${t}`;
    }).join('\n');
    const codingKeys = Object.entries(obj)
      .filter(([k]) => k !== toCamelCase(k))
      .map(([k]) => `        case ${toCamelCase(k)} = "${k}"`)
      .join('\n');
    const ck = codingKeys ? `\n\n    enum CodingKeys: String, CodingKey {\n${codingKeys}\n    }` : '';
    structs.push(`struct ${name}: Codable {\n${fields}${ck}\n}`);
  }

  genStruct(Array.isArray(parsed) ? parsed[0] : parsed, rootName);
  setOutput(structs.reverse().join('\n\n'), 'swift');
}
```

### Rust（`/static/js/json-tool-to-rust.js`）

```javascript
function processJson() {
  const parsed   = parseInput();
  if (parsed === null) return;
  const rootName = toPascalCase(document.getElementById('className')?.value || 'Root');
  const structs  = [];

  function genStruct(obj, name) {
    if (typeof obj !== 'object' || obj === null || Array.isArray(obj)) return;
    const fields = Object.entries(obj).map(([k, v]) => {
      const snakeKey = k.replace(/([A-Z])/g, '_$1').toLowerCase();
      const rename   = snakeKey !== k ? `    #[serde(rename = "${k}")]\n` : '';
      if (typeof v === 'object' && v !== null && !Array.isArray(v)) {
        genStruct(v, toPascalCase(k));
        return `${rename}    pub ${snakeKey}: ${toPascalCase(k)},`;
      }
      if (Array.isArray(v) && v.length > 0 && typeof v[0] === 'object') {
        genStruct(v[0], toPascalCase(k));
        return `${rename}    pub ${snakeKey}: Vec<${toPascalCase(k)}>,`;
      }
      const rustType = v === null ? 'Option<serde_json::Value>'
        : typeof v === 'boolean' ? 'bool'
        : typeof v === 'number'  ? (Number.isInteger(v) ? 'i64' : 'f64')
        : 'String';
      return `${rename}    pub ${snakeKey}: ${rustType},`;
    }).join('\n');
    structs.push(`#[derive(Debug, Serialize, Deserialize)]\npub struct ${name} {\n${fields}\n}`);
  }

  genStruct(Array.isArray(parsed) ? parsed[0] : parsed, rootName);
  const imports = 'use serde::{Deserialize, Serialize};\n\n';
  setOutput(imports + structs.reverse().join('\n\n'), 'rust');
}
```

### PHP（`/static/js/json-tool-to-php.js`）

```javascript
function processJson() {
  const parsed   = parseInput();
  if (parsed === null) return;
  const rootName = toPascalCase(document.getElementById('className')?.value || 'Root');

  function genClass(obj, name) {
    if (typeof obj !== 'object' || obj === null || Array.isArray(obj)) return '';
    const props = Object.entries(obj).map(([k, v]) => {
      const phpType = v === null ? '' : typeof v === 'boolean' ? 'bool ' : typeof v === 'number' ? (Number.isInteger(v)?'int ':'float ') : 'string ';
      return `    public ${phpType}$${k};`;
    }).join('\n');
    return `class ${name} {\n${props}\n\n    public static function fromArray(array $data): self {\n        $obj = new self();\n        foreach ($data as $key => $value) {\n            $obj->$key = $value;\n        }\n        return $obj;\n    }\n}`;
  }

  const src = Array.isArray(parsed) ? parsed[0] : parsed;
  setOutput(`<?php\n\n${genClass(src, rootName)}\n`, 'php');
}
```
