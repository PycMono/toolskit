```javascript
// /static/js/json-tools-core.js
// Monaco Editor 初始化 + 公共工具函数

'use strict';

let inputEditor  = null;
let outputEditor = null;

// Monaco 初始化
require.config({ paths: { vs: 'https://cdnjs.cloudflare.com/ajax/libs/monaco-editor/0.45.0/min/vs' }});

require(['vs/editor/editor.main'], function() {
  inputEditor = monaco.editor.create(document.getElementById('inputEditor'), {
    value:          '',
    language:       'json',
    theme:          'vs',
    fontSize:       13,
    minimap:        { enabled: false },
    scrollBeyondLastLine: false,
    automaticLayout:true,
    wordWrap:       'on',
  });

  outputEditor = monaco.editor.create(document.getElementById('outputEditor'), {
    value:          '',
    language:       'json',
    theme:          'vs',
    fontSize:       13,
    minimap:        { enabled: false },
    readOnly:       true,
    scrollBeyondLastLine: false,
    automaticLayout:true,
    wordWrap:       'on',
  });

  inputEditor.onDidChangeModelContent(() => {
    updateInputStats();
  });

  // 初始化时从 URL hash 恢复（分享链接场景）
  const hash = location.hash.slice(1);
  if (hash) {
    try { inputEditor.setValue(decodeURIComponent(hash)); } catch(e) {}
  }
});

function getInput()  { return inputEditor?.getValue()  || ''; }
function getOutput() { return outputEditor?.getValue() || ''; }
function setOutput(val, lang) {
  outputEditor?.setValue(val);
  if (lang) {
    monaco.editor.setModelLanguage(outputEditor.getModel(), lang);
  }
  updateOutputStats(val);
}

function parseInput() {
  const raw = getInput().trim();
  if (!raw) { showToast(window.JT_I18N?.['json.common.error.empty'] || '请先输入 JSON', 'error'); return null; }
  try {
    return JSON.parse(raw);
  } catch(e) {
    showErrorPanel(e, raw);
    return null;
  }
}

function loadExample() {
  const example = {
    "name": "John Doe",
    "age": 30,
    "email": "john@example.com",
    "address": {
      "city": "Beijing",
      "country": "China"
    },
    "hobbies": ["coding", "reading", "hiking"],
    "scores": [98.5, 87, 92.3]
  };
  inputEditor?.setValue(JSON.stringify(example, null, 2));
}

function clearInput()  { inputEditor?.setValue('');  clearErrorPanel(); }
function clearOutput() { outputEditor?.setValue(''); }

function copyOutput() {
  const text = getOutput();
  if (!text) return;
  navigator.clipboard.writeText(text).then(() => {
    showToast(window.JT_I18N?.['json.common.copy_done'] || '已复制！', 'success');
  });
}

function downloadOutput(filename, ext) {
  const text = getOutput();
  if (!text) return;
  const blob = new Blob([text], { type: 'text/plain;charset=utf-8' });
  saveAs(blob, filename || `output.${ext || 'json'}`);
}

function swapEditors() {
  const inp = getInput();
  const out = getOutput();
  inputEditor?.setValue(out);
  outputEditor?.setValue(inp); // 临时取消 readOnly
}

function uploadFile(input) {
  const file = input.files[0];
  if (!file) return;
  const reader = new FileReader();
  reader.onload = e => inputEditor?.setValue(e.target.result);
  reader.readAsText(file, 'utf-8');
  input.value = '';
}

function updateInputStats() {
  const text = getInput();
  const el   = document.getElementById('inputSize');
  if (el) el.textContent = formatBytes(new Blob([text]).size);
}

function updateOutputStats(text) {
  const el = document.getElementById('outputSize');
  if (el) el.textContent = formatBytes(new Blob([text || '']).size);
}

// 错误面板
function showErrorPanel(err, raw) {
  const panel = document.getElementById('errorPanel');
  const title = document.getElementById('errorTitle');
  const body  = document.getElementById('errorBody');
  if (!panel) return;

  const analysis = analyzeJsonError(err, raw);

  title.textContent = analysis.summary;
  body.innerHTML = `
    <div class="jt-error-detail">
      <div class="jt-error-location">
        📍 第 <strong>${analysis.line}</strong> 行，第 <strong>${analysis.col}</strong> 列
      </div>
      <div class="jt-error-message">${analysis.message}</div>
      ${analysis.suggestion ? `
      <div class="jt-error-suggestion">
        💡 修复建议：${analysis.suggestion}
      </div>` : ''}
      ${analysis.context ? `
      <pre class="jt-error-context">${escapeHtml(analysis.context)}</pre>
      ` : ''}
    </div>
  `;

  panel.style.display = 'block';

  // 高亮错误行
  if (analysis.line && inputEditor) {
    inputEditor.deltaDecorations([], [{
      range: new monaco.Range(analysis.line, 1, analysis.line, 9999),
      options: { isWholeLine: true, className: 'jt-error-line' },
    }]);
    inputEditor.revealLineInCenter(analysis.line);
  }
}

function hideErrorPanel() {
  const panel = document.getElementById('errorPanel');
  if (panel) panel.style.display = 'none';
  inputEditor?.deltaDecorations(
    inputEditor.getModel().getAllDecorations().map(d => d.id), []
  );
}
function clearErrorPanel() { hideErrorPanel(); }

// JSON 错误智能分析
function analyzeJsonError(err, raw) {
  const msg  = err.message || '';
  const lines = raw.split('\n');
  let line = 1, col = 1;

  // 从错误信息中提取位置（Chrome/Firefox 格式）
  const posMatch = msg.match(/position (\d+)/i);
  if (posMatch) {
    const pos = parseInt(posMatch[1]);
    let count = 0;
    for (let i = 0; i < lines.length; i++) {
      if (count + lines[i].length + 1 > pos) {
        line = i + 1;
        col  = pos - count + 1;
        break;
      }
      count += lines[i].length + 1;
    }
  }

  // 提取错误上下文（错误行前后各1行）
  const ctxLines = lines.slice(Math.max(0, line-2), line+1);
  const context  = ctxLines.join('\n');

  // 智能错误建议
  const near = lines[line - 1]?.substring(0, col + 5) || '';
  let suggestion = '';
  if (/,\s*[}\]]/.test(raw))            suggestion = '移除尾随逗号（最后一个元素后不能有逗号）';
  else if (/[^\\]'/.test(near))         suggestion = '使用双引号 " 代替单引号 \'';
  else if (/:\s*undefined/.test(near))  suggestion = 'undefined 不是合法 JSON 值，请使用 null';
  else if (/\/\//.test(raw))            suggestion = 'JSON 不支持注释，请使用 /json/jsonc 工具先处理';
  else if (/{[^}]*$/.test(raw))         suggestion = '对象括号 { 未闭合';
  else if (/\[[^\]]*$/.test(raw))       suggestion = '数组括号 [ 未闭合';

  return {
    summary:    '❌ JSON 格式错误',
    line, col,
    message:    msg,
    suggestion,
    context,
  };
}

// 通用工具函数
function formatBytes(bytes) {
  if (bytes < 1024)        return `${bytes} B`;
  if (bytes < 1024*1024)   return `${(bytes/1024).toFixed(1)} KB`;
  return `${(bytes/1024/1024).toFixed(2)} MB`;
}

function escapeHtml(str) {
  return String(str).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');
}

function showToast(msg, type = 'info') {
  const container = document.getElementById('toastContainer');
  if (!container) return;
  const el = document.createElement('div');
  el.className   = `jt-toast jt-toast--${type}`;
  el.textContent = msg;
  container.appendChild(el);
  requestAnimationFrame(() => el.classList.add('jt-toast--show'));
  setTimeout(() => {
    el.classList.remove('jt-toast--show');
    setTimeout(() => el.remove(), 300);
  }, 3000);
}

function toggleFAQ(id) {
  const item   = document.getElementById(id);
  const isOpen = item.classList.contains('jt-faq-item--open');
  document.querySelectorAll('.jt-faq-item--open').forEach(i => i.classList.remove('jt-faq-item--open'));
  if (!isOpen) item.classList.add('jt-faq-item--open');
}
```

```javascript
function processJson() {
  const raw = getInput().trim();
  if (!raw) return;
  clearErrorPanel();
  try {
    const parsed = JSON.parse(raw);
    const indent = parseInt(document.getElementById('indentSize')?.value || 2);
    setOutput(JSON.stringify(parsed, null, indent));
    showSuccessBanner();
  } catch(e) {
    showErrorPanel(e, raw);
    setOutput('');
  }
}

function showSuccessBanner() {
  const el = document.getElementById('outputStats');
  if (el) {
    el.innerHTML = '<span class="jt-success-badge">✅ JSON 合法</span>';
  }
}
```

### minify 工具（`/static/js/json-tool-minify.js`）

```javascript
function processJson() {
  const parsed = parseInput();
  if (parsed === null) return;
  const minified = JSON.stringify(parsed);
  setOutput(minified);
  const original = new Blob([getInput()]).size;
  const compact  = new Blob([minified]).size;
  const saved    = Math.round((1 - compact/original)*100);
  const el = document.getElementById('outputStats');
  if (el) el.textContent = `节省 ${saved}%（${formatBytes(original)} → ${formatBytes(compact)}）`;
}
```

### sort-keys 工具（`/static/js/json-tool-sort-keys.js`）

```javascript
function processJson() {
  const parsed = parseInput();
  if (parsed === null) return;
  const order = document.querySelector('input[name="sortOrder"]:checked')?.value || 'asc';
  setOutput(JSON.stringify(sortKeys(parsed, order), null, 2));
}

function sortKeys(obj, order) {
  if (Array.isArray(obj))      return obj.map(v => sortKeys(v, order));
  if (typeof obj !== 'object' || obj === null) return obj;
  const keys = Object.keys(obj).sort((a, b) =>
    order === 'asc' ? a.localeCompare(b) : b.localeCompare(a)
  );
  const result = {};
  for (const k of keys) result[k] = sortKeys(obj[k], order);
  return result;
}
```
