## 2. Diff 工具 JS（`/static/js/json-tool-diff.js`）

```javascript
function processJson() {
  const leftRaw  = document.getElementById('diffLeftEditor')  ? leftEditor.getValue()  : '';
  const rightRaw = document.getElementById('diffRightEditor') ? rightEditor.getValue() : '';

  let left, right;
  try { left  = JSON.parse(leftRaw);  } catch(e) { showToast('左侧 JSON 格式错误', 'error'); return; }
  try { right = JSON.parse(rightRaw); } catch(e) { showToast('右侧 JSON 格式错误', 'error'); return; }

  const diffs = diffJson(left, right);
  renderDiffResult(diffs);
}

// 递归比较两个 JSON 对象
function diffJson(a, b, path = '$') {
  const results = [];

  if (JSON.stringify(a) === JSON.stringify(b)) return results;

  if (typeof a !== typeof b || Array.isArray(a) !== Array.isArray(b)) {
    results.push({ type: 'changed', path, before: a, after: b });
    return results;
  }

  if (typeof a !== 'object' || a === null) {
    if (a !== b) results.push({ type: 'changed', path, before: a, after: b });
    return results;
  }

  const allKeys = new Set([...Object.keys(a), ...Object.keys(b)]);
  for (const k of allKeys) {
    const childPath = Array.isArray(a) ? `${path}[${k}]` : `${path}.${k}`;
    if (!(k in a)) {
      results.push({ type: 'added', path: childPath, after: b[k] });
    } else if (!(k in b)) {
      results.push({ type: 'removed', path: childPath, before: a[k] });
    } else {
      results.push(...diffJson(a[k], b[k], childPath));
    }
  }
  return results;
}

function renderDiffResult(diffs) {
  const container = document.getElementById('diffResult');
  if (diffs.length === 0) {
    container.innerHTML = '<div class="jt-diff-same">✅ 两个 JSON 完全相同</div>';
    return;
  }

  const summary = {
    added:   diffs.filter(d => d.type === 'added').length,
    removed: diffs.filter(d => d.type === 'removed').length,
    changed: diffs.filter(d => d.type === 'changed').length,
  };

  container.innerHTML = `
    <div class="jt-diff-summary">
      <span class="jt-diff-badge jt-diff-badge--added">+${summary.added} 新增</span>
      <span class="jt-diff-badge jt-diff-badge--removed">-${summary.removed} 删除</span>
      <span class="jt-diff-badge jt-diff-badge--changed">~${summary.changed} 修改</span>
    </div>
    ${diffs.map(d => `
      <div class="jt-diff-item jt-diff-item--${d.type}">
        <span class="jt-diff-path">${escapeHtml(d.path)}</span>
        ${d.before !== undefined ? `<del class="jt-diff-before">${escapeHtml(JSON.stringify(d.before))}</del>` : ''}
        ${d.after  !== undefined ? `<ins class="jt-diff-after">${escapeHtml(JSON.stringify(d.after))}</ins>` : ''}
      </div>
    `).join('')}
  `;
}
```

## 3. JSON Path 工具 JS（`/static/js/json-tool-path.js`）

```javascript
// 依赖：jsonpath-plus CDN

function processJson() {
  const parsed = parseInput();
  if (parsed === null) return;

  const expression = document.getElementById('pathInput')?.value.trim();
  if (!expression) { showToast('请输入 JSONPath 表达式', 'error'); return; }

  try {
    const { JSONPath } = window;
    const results = JSONPath({ path: expression, json: parsed });
    setOutput(JSON.stringify(results, null, 2));
    const el = document.getElementById('outputStats');
    if (el) el.textContent = `匹配 ${results.length} 条结果`;
  } catch(e) {
    showToast('JSONPath 表达式错误：' + e.message, 'error');
  }
}
```

## 4. JSON Search 工具 JS（`/static/js/json-tool-search.js`）

```javascript
function processJson() {
  const parsed = parseInput();
  if (parsed === null) return;

  const query   = document.getElementById('searchInput')?.value.trim();
  if (!query) return;

  const useRegex = document.getElementById('useRegex')?.checked;
  const searchIn = document.querySelector('input[name="searchIn"]:checked')?.value || 'both';

  const results = searchJson(parsed, query, useRegex, searchIn, '$');
  renderSearchResults(results);
}

function searchJson(node, query, useRegex, searchIn, path) {
  const results = [];
  const matcher = useRegex
    ? (s) => new RegExp(query, 'i').test(String(s))
    : (s) => String(s).toLowerCase().includes(query.toLowerCase());

  function walk(v, p) {
    if (typeof v === 'object' && v !== null) {
      for (const [k, child] of Object.entries(v)) {
        const childPath = Array.isArray(v) ? `${p}[${k}]` : `${p}.${k}`;
        if ((searchIn === 'key' || searchIn === 'both') && matcher(k)) {
          results.push({ path: childPath, matchType: 'key', key: k, value: child });
        }
        walk(child, childPath);
      }
    } else {
      if ((searchIn === 'value' || searchIn === 'both') && matcher(v)) {
        results.push({ path: p, matchType: 'value', value: v });
      }
    }
  }

  walk(node, path);
  return results;
}

function renderSearchResults(results) {
  const container = document.getElementById('searchOutput');
  if (results.length === 0) {
    container.innerHTML = '<p class="jt-search-empty">未找到匹配项</p>';
    return;
  }
  container.innerHTML = `
    <p class="jt-search-count">找到 ${results.length} 处匹配</p>
    ${results.map(r => `
      <div class="jt-search-item">
        <span class="jt-search-path">${escapeHtml(r.path)}</span>
        <span class="jt-search-badge jt-search-badge--${r.matchType}">${r.matchType === 'key' ? 'Key' : 'Value'}</span>
        <span class="jt-search-value">${escapeHtml(JSON.stringify(r.value))}</span>
      </div>
    `).join('')}
  `;
}
```
